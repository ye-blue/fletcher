61 void SchemaSet::AppendSchema(const std::shared_ptr<arrow::Schema>& arrow_schema) {
62  auto name = fletcher::GetMeta(*arrow_schema, "fletcher_name");
63  if (name.empty()) {
64  FLETCHER_LOG(WARNING, "Skipping anonymous schema with the following contents:\n" + arrow_schema->ToString());
65  FLETCHER_LOG(WARNING, "Append {'fletcher_name' : '<name>'} kv-metadata to the schema "
66  "to include this schema in hardware generation.");
67  return;
68  }
69  if (!HasSchemaWithName(name)) {
70  auto fs = std::make_shared<FletcherSchema>(arrow_schema);
71  schemas_.push_back(fs);
72  } else {
73  // If a schema with this name already exists. If so, check if its equal.
74  auto other = GetSchema(name);
75  if ((*other)->arrow_schema()->Equals(*arrow_schema)) {
76  FLETCHER_LOG(INFO, "Duplicate but equal schema in SchemaSet: " + name);
77  }
78  }
79 }

109 void SchemaSet::Sort() {
110  struct {
111  bool operator()(const std::shared_ptr<FletcherSchema> &a, const std::shared_ptr<FletcherSchema> &b) {
112  return a->name() < b->name();
113  }
114  } NameSort;
115 
116  struct {
117  bool operator()(const std::shared_ptr<FletcherSchema> &a, const std::shared_ptr<FletcherSchema> &b) {
118  if (a->mode() == b->mode()) {
119  return false;
120  } else if (a->mode() == Mode::WRITE) {
121  return false;
122  }
123  return true;
124  }
125  } ModeSort;
126 
127  // Sort the schemas by name, then by mode
128  std::sort(schemas_.begin(), schemas_.end(), NameSort);
129  std::stable_sort(schemas_.begin(), schemas_.end(), ModeSort);
130 }

132 FletcherSchema::FletcherSchema(const std::shared_ptr<arrow::Schema> &arrow_schema, const std::string &schema_name)
133  : arrow_schema_(arrow_schema), mode_(fletcher::GetMode(*arrow_schema)) {
134  // Get name from metadata, if available

136  if (name_.empty()) {
