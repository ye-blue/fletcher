33 RecordBatch::RecordBatch(const std::shared_ptr<FletcherSchema> &fletcher_schema)
34  : Component(fletcher_schema->name()), fletcher_schema_(fletcher_schema) {
35  // Get Arrow Schema
36  auto as = fletcher_schema_->arrow_schema();
37  // Add default port nodes
38  Add(Port::Make("bcd", cr(), Port::Dir::IN, bus_cd()));
39  Add(Port::Make("kcd", cr(), Port::Dir::IN, kernel_cd()));
40  Add(bus_addr_width());
41  // Add and connect all array readers and resulting ports
42  AddArrays(fletcher_schema);

45 void RecordBatch::AddArrays(const std::shared_ptr<FletcherSchema> &fletcher_schema) {
46  // Iterate over all fields and add ArrayReader/Writer data and control ports.
47  for (const auto &field : fletcher_schema->arrow_schema()->fields()) {
48  // Check if we must ignore the field
49  if (fletcher::MustIgnore(*field)) {
50  FLETCHER_LOG(DEBUG, "Ignoring field " + field->name());
51  } else {
52  FLETCHER_LOG(DEBUG, "Instantiating Array" << (fletcher_schema->mode() == Mode::READ ? "Reader" : "Writer")
53  << " for schema: " << fletcher_schema->name()
54  << " : " << field->name());
55  // Generate a warning for Writers as they are still experimental.
56  if (fletcher_schema->mode() == Mode::WRITE) {
57  FLETCHER_LOG(WARNING, "ArrayWriter implementation is highly experimental. Use with caution! Features that are "
58  "not implemented include:\n"
59  " - dvalid bit is ignored (so you cannot supply handshakes on the values stream for "
60  "empty lists or use empty handshakes to close streams)\n"
61  " - lists of primitives (e.g. strings) values stream last signal must signal the last "
62  "value for all lists, not single lists in the Arrow Array).\n"
63  " - clock domain crossings.");
64  }
65 
66  // Convert to and add an Arrow port. We must invert it because it is an output of the RecordBatch.
67  auto kernel_arrow_port = FieldPort::MakeArrowPort(fletcher_schema,
68  field,
69  fletcher_schema->mode(),
70  true,
71  kernel_cd());
72  auto kernel_arrow_type = kernel_arrow_port->type();
73  Add(kernel_arrow_port);
74 
75  // Add a command port for the ArrayReader/Writer
76  auto command_port = FieldPort::MakeCommandPort(fletcher_schema, field, true, kernel_cd());
77  Add(command_port);
78 
79  // Add an unlock port for the ArrayReader/Writer
80  auto unlock_port = FieldPort::MakeUnlockPort(fletcher_schema, field, kernel_cd());
81  Add(unlock_port);
82 
83  // Instantiate an ArrayReader or Writer
84  auto array_inst_unique = ArrayInstance(field->name() + "_inst",
85  fletcher_schema->mode(),
86  kernel_arrow_port->data_width(),
87  ctrl_width(*field),
88  tag_width(*field));
89  auto array_inst = array_inst_unique.get(); // Remember the raw pointer
90  array_instances_.push_back(array_inst);
91  AddChild(std::move(array_inst_unique)); // Move ownership to this RecordBatchReader/Writer.
92 
93  // Generate a configuration string for the ArrayReader
94  auto cfg_node = array_inst->GetNode(Node::NodeID::PARAMETER, "CFG");
95  cfg_node <<= cerata::strl(GenerateConfigString(*field)); // Set the configuration string for this field
96 
97  // Drive the clocks and resets
98  Connect(array_inst->port("kcd"), port("kcd"));
99  Connect(array_inst->port("bcd"), port("bcd"));
100 
101  // Drive the RecordBatch Arrow data port with the ArrayReader/Writer data port, or vice versa
102  if (fletcher_schema->mode() == Mode::READ) {
103  auto array_data_port = array_inst->port("out");
104  auto array_data_type = array_data_port->type();
105  // Create a mapper between the Arrow port and the Array data port
106  auto mapper = GetStreamTypeMapper(kernel_arrow_type, array_data_type);
107  kernel_arrow_type->AddMapper(mapper);
108  // Connect the ports
109  kernel_arrow_port <<= array_data_port;
110  } else {
111  auto array_data_port = array_inst->port("in");
112  auto array_data_type = array_data_port->type();
113  // Create a mapper between the Arrow port and the Array data port
114  auto mapper = GetStreamTypeMapper(kernel_arrow_type, array_data_type);
115  kernel_arrow_type->AddMapper(mapper);
116  // Connect the ports
117  array_data_port <<= kernel_arrow_port;
118  }
119 
120  // Drive the ArrayReader command port from the top-level command port
121  array_inst->port("cmd") <<= command_port;
122  // Drive the unlock port with the ArrayReader/Writer
123  unlock_port <<= array_inst->port("unl");
124 
125  // Copy over the ArrayReader/Writer's bus channels
126  auto bus = std::dynamic_pointer_cast<BusPort>(array_inst->port("bus")->Copy());
127  // Give the new bus port a unique name
128  // TODO(johanpel): move the bus renaming to the Mantle level
129  bus->SetName(fletcher_schema->name() + "_" + field->name() + "_" + bus->name());
130  Add(bus); // Add them to the RecordBatch
131  bus_ports_.push_back(bus); // Remember the port
132  bus <<= array_inst->port("bus"); // Connect them to the ArrayReader/Writer
133  }
134  }

137 std::shared_ptr<FieldPort> RecordBatch::GetArrowPort(const arrow::Field &field) const {
138  for (const auto &n : objects_) {
139  auto ap = std::dynamic_pointer_cast<FieldPort>(n);
140  if (ap != nullptr) {
141  if ((ap->function_ == FieldPort::ARROW) && (ap->field_->Equals(field))) {
142  return ap;
143  }
144  }
145  }
146  throw std::runtime_error("Field " + field.name() + " did not generate an ArrowPort for Core " + name() + ".");
147 }

150 RecordBatch::GetFieldPorts(const std::optional<FieldPort::Function> &function) const {
151  std::deque<std::shared_ptr<FieldPort>> result;
152  for (const auto &n : objects_) {
153  auto ap = std::dynamic_pointer_cast<FieldPort>(n);
154  if (ap != nullptr) {
155  if ((function && (ap->function_ == *function)) || !function) {
156  result.push_back(ap);
157  }
158  }
159  }
160  return result;

170 std::shared_ptr<FieldPort> FieldPort::MakeArrowPort(const std::shared_ptr<FletcherSchema> &fletcher_schema,
171  const std::shared_ptr<arrow::Field> &field,
172  Mode mode,
173  bool invert,
174  const std::shared_ptr<ClockDomain> &domain) {
175  Dir dir;
176  if (invert) {
177  dir = Term::Invert(mode2dir(mode));
178  } else {
179  dir = mode2dir(mode);
180  }
181  return std::make_shared<FieldPort>(fletcher_schema->name() + "_" + field->name(),
182  ARROW, field, fletcher_schema, GetStreamType(*field, mode), dir, domain);
183 }

185 std::shared_ptr<FieldPort> FieldPort::MakeCommandPort(const std::shared_ptr<FletcherSchema> &fletcher_schema,
186  const std::shared_ptr<arrow::Field> &field,
187  bool ctrl,
188  const std::shared_ptr<ClockDomain> &domain) {
189  std::shared_ptr<cerata::Type> type;
190  if (ctrl) {
191  type = cmd(tag_width(*field), ctrl_width(*field));
192  } else {
193  type = cmd(tag_width(*field));
194  }
195  return std::make_shared<FieldPort>(fletcher_schema->name() + "_" + field->name() + "_cmd",
196  COMMAND,
197  field,
198  fletcher_schema,
199  type,
200  Dir::IN,
201  domain);
202 }

216 std::shared_ptr<Node> FieldPort::data_width() {
217  std::shared_ptr<Node> width = intl(0);
218  // Flatten the type
219  auto flat_type = Flatten(type());
220  for (const auto &ft : flat_type) {
221  if (ft.type_->meta.count("array_data") > 0) {
222  if (ft.type_->width()) {
223  width = width + ft.type_->width().value();
224  }
225  }
226  }
227  return width;
228 }
