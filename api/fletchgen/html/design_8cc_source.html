34 static std::optional<std::shared_ptr<arrow::RecordBatch>> GetRecordBatchWithName(const RBVector &batches,
35  const std::string &name) {
36  for (const auto &b : batches) {
37  if (fletcher::GetMeta(*b->schema(), "fletcher_name") == name) {
38  return b;

44 void Design::AnalyzeSchemas() {
45  // Attempt to create a SchemaSet from all schemas that can be detected in the options.
46  FLETCHER_LOG(INFO, "Creating SchemaSet.");
47  schema_set = SchemaSet::Make(options->kernel_name);
48  // Add all schemas from the list of schema files
49  for (const auto &arrow_schema : options->schemas) {
50  schema_set->AppendSchema(arrow_schema);
51  }
52  // Add all schemas from the recordbatches and add all recordbatches.
53  for (const auto &recordbatch : options->recordbatches) {
54  schema_set->AppendSchema(recordbatch->schema());
55  }
56 
57  // Sort the schema set according to the recordbatch ordering specification.
58  // Important for the control flow through MMIO / buffer addresses.
59  // First we sort recordbatches by name, then by mode.
60  schema_set->Sort();
61 }

63 void Design::AnalyzeRecordBatches() {
64  // Now that we have every Schema, for every Schema, figure out if there is a RecordBatch in the input options.
65  // If there is, add a description of the RecordBatch to this design.
66  // If there isn't, create a virtual RecordBatch based on the schema.
67  for (const auto &fletcher_schema : schema_set->schemas()) {
68  auto rb = GetRecordBatchWithName(options->recordbatches, fletcher_schema->name());
69  fletcher::RecordBatchDescription rbd;
70  if (rb) {
71  fletcher::RecordBatchAnalyzer rba(&rbd);
72  rba.Analyze(**rb);
73  } else {
74  fletcher::SchemaAnalyzer sa(&rbd);
75  sa.Analyze(*fletcher_schema->arrow_schema());
76  }
77  batch_desc.push_back(rbd);
78  }
79 }

81 static std::vector<MmioReg> ParseRegs(const std::vector<std::string> &regs) {
82  std::vector<MmioReg> result;
83 
84  for (const auto &reg_str : regs) {
85  std::regex expr(R"([c|s][\:][\d]+[\:][\w]+)");
86  if (std::regex_match(reg_str, expr)) {
87  MmioReg reg;
88  auto w_start = reg_str.find(':') + 1;
89  auto i_start = reg_str.find(':', w_start) + 1;
90  auto width_str = reg_str.substr(w_start, i_start - w_start);
91  reg.name = reg_str.substr(i_start);
92  reg.width = static_cast<uint32_t>(std::strtoul(width_str.c_str(), nullptr, 10));
93  switch (reg_str[0]) {
94  case 'c':reg.behavior = MmioReg::Behavior::CONTROL;
95  break;
96  case 's':reg.behavior = MmioReg::Behavior::STATUS;
97  break;
98  default:FLETCHER_LOG(FATAL, "Register argument behavior character invalid for " + reg.name);
99  }
100  // Calculate how much address space this register needs by rounding up to AXI4-lite words,
101  // that are byte addressed.
102  reg.addr_space_used = 4 * (reg.width / 32 + (reg.width % 32 != 0));
103  result.push_back(reg);
104  }
105  }
106  return result;
107 }

109 fletchgen::Design fletchgen::Design::GenerateFrom(const std::shared_ptr<Options> &opts) {
110  Design ret;
111  ret.options = opts;
112 
113  // Analyze schemas and recordbatches to get schema_set and batch_desc
114  ret.AnalyzeSchemas();
115  ret.AnalyzeRecordBatches();
116 
117  // Generate the hardware structure through Mantle and extract all subcomponents.
118  FLETCHER_LOG(INFO, "Generating Mantle...");
119  ret.custom_regs = ParseRegs(opts->regs);
120  ret.mantle = Mantle::Make(*ret.schema_set, ret.batch_desc, ret.custom_regs);
121  ret.kernel = ret.mantle->nucleus()->kernel;
122  ret.nucleus = ret.mantle->nucleus();
123  for (const auto &recordbatch_component : ret.mantle->recordbatch_components()) {
124  ret.recordbatches.push_back(recordbatch_component);
125  }
126 
127  // Generate a Yaml file for vhdmmio based on the recordbatch description
128  GenerateVhdmmioYaml(ret.batch_desc, ret.custom_regs);
129 
130  // TODO(johanpel): run vhdmmio in a nicer way
131  // Run vhdmmio
132  auto vhdmmio_result = system("vhdmmio -V vhdl -H -P vhdl");
133  if (vhdmmio_result != 0) {
134  FLETCHER_LOG(FATAL, "vhdmmio exited with status " << vhdmmio_result);
135  }
136 
137  return ret;
138 }

140 std::deque<cerata::OutputSpec> Design::GetOutputSpec() {
141  std::deque<OutputSpec> result;
142  OutputSpec omantle, okernel, onucleus;
143 
144  // Mantle
145  omantle.comp = mantle;
146  // Always overwrite mantle, as users should not modify.
147  omantle.meta[cerata::vhdl::metakeys::OVERWRITE_FILE] = "true";
148  result.push_back(omantle);
149 
150  // Nucleus
151  onucleus.comp = nucleus;
152  // Check the force flag if kernel should be overwritten
153  onucleus.meta[cerata::vhdl::metakeys::OVERWRITE_FILE] = "true";
154  result.push_back(onucleus);
155 
156  // Kernel
157  okernel.comp = kernel;
158  // Check the force flag if kernel should be overwritten
159  okernel.meta[cerata::vhdl::metakeys::OVERWRITE_FILE] = options->overwrite ? "true" : "false";
160  result.push_back(okernel);
161 
162  // RecordBatchReaders/Writers
163  for (const auto &recbatch : recordbatches) {
164  OutputSpec orecbatch;
165  orecbatch.comp = recbatch;
166  // Always overwrite readers/writers, as users should not modify.
167  orecbatch.meta[cerata::vhdl::metakeys::OVERWRITE_FILE] = "true";
168  result.push_back(orecbatch);
169  }
170 
171  return result;
172 }
