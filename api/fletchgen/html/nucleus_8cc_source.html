28 ArrayCmdCtrlMerger::ArrayCmdCtrlMerger() : Component("ArrayCmdCtrlMerger") {
29  // This is a primitive component from the hardware lib
30  meta_[cerata::vhdl::metakeys::PRIMITIVE] = "true";
31  meta_[cerata::vhdl::metakeys::LIBRARY] = "work";
32  meta_[cerata::vhdl::metakeys::PACKAGE] = "Array_pkg";

34  auto reg64 = cerata::Vector::Make(64);
35  auto baw = Parameter::Make("bus_addr_width", integer(), intl(64));
36  auto idw = Parameter::Make("index_width", integer(), intl(32));
37  auto tw = Parameter::Make("tag_width", integer(), intl(1));
38  auto cw = Parameter::Make("num_addr", integer(), intl(0));

41  auto ctrl = PortArray::Make("ctrl", reg64, cw, Port::Dir::IN, kernel_cd());
42  Add({baw, idw, tw, nucleus_side_cmd, kernel_side_cmd, ctrl});

62 static void CopyFieldPorts(Component *nucleus,
63  Component *kernel,
64  const RecordBatch &record_batch,
65  FieldPort::Function fun) {
66  // Add Arrow field derived ports with some function.
67  auto field_ports = record_batch.GetFieldPorts(fun);
68  for (const auto &fp : field_ports) {
69  // Create a copy and invert for the Nucleus
70  auto copied_port = std::dynamic_pointer_cast<FieldPort>(fp->Copy());
71  copied_port->InvertDirection();
72  nucleus->Add(copied_port);
73  // Create and add another copy for the kernel component
74  kernel->Add(copied_port->Copy());
75  }
76 }

78 Nucleus::Nucleus(const std::string &name,
79  const std::deque<RecordBatch *> &recordbatches,
80  const std::vector<fletcher::RecordBatchDescription> &batch_desc,
81  const std::vector<MmioReg> &custom_regs)
82  : Component("Nucleus_" + name) {
83 
84  // TODO(johanpel): move this sanity check to where it should be
85  if (recordbatches.size() != batch_desc.size()) {
86  throw std::runtime_error("Number of recordbatch components not equal to recordbatch descriptors.");
87  }
88 
89  // Add address width
90  Add(bus_addr_width());
91 
92  // Add clock/reset
93  auto kcd = Port::Make("kcd", cr(), Port::Dir::IN, kernel_cd());
94  Add(kcd);
95 
96  // Create kernel
97  kernel = Kernel::Make(name, this);
98 
99  // Add MMIO port
100  auto mmio = MmioPort::Make(Port::Dir::IN);
101  Add(mmio);
102 
103  // Add VHDMMIO component
104  std::vector<std::string> vhdmmio_buf_port_names;
105  auto vhdmmio = GenerateMmioComponent(batch_desc, custom_regs, &vhdmmio_buf_port_names);
106  auto vhdmmio_inst = AddInstanceOf(vhdmmio.get());
107 
108  vhdmmio_inst->port("mmio") <<= mmio;
109  vhdmmio_inst->port("kcd") <<= kcd;
110 
111  // Add ports from mmio
112  for (const auto &p : vhdmmio_inst->GetAll<Port>()) {
113  if (p->meta.count(MMIO_KERNEL) > 0) {
114  auto name_intended = p->meta.at(MMIO_KERNEL);
115  auto kernel_mmio_port = std::dynamic_pointer_cast<Port>(p->Copy());
116  kernel_mmio_port->InvertDirection();
117  kernel_mmio_port->SetName(name_intended);
118  kernel->Add(kernel_mmio_port);
119  }
120  }
121 
122  std::vector<Instance *> accms;
123 
124  // Add ports going to/from RecordBatches
125  for (auto r : recordbatches) {
126  // Just copy over the Arrow data and unlock stream ports. Their fields are not abstracted for the kernel user.
127  CopyFieldPorts(this, kernel.get(), *r, FieldPort::Function::ARROW);
128  CopyFieldPorts(this, kernel.get(), *r, FieldPort::Function::UNLOCK);
129 
130  // For the kernel user, we need to abstract the "ctrl" field of the command stream away.
131  // Because TODO(johanpel): flat type slicing isn't implemented
132  // we need to instantiate a little component that just adds the buffer addresses to the ctrl stream.
133 
134  // First, obtain all command stream ports.
135  auto cmd_ports = r->GetFieldPorts(FieldPort::Function::COMMAND);
136  for (auto &cp : cmd_ports) {
137  // For each one of them, make an inverted copy of the recordbatch unabstracted command stream port. This one will
138  // expose all command stream fields to the nucleus user.
139  auto nucleus_cmd = std::dynamic_pointer_cast<FieldPort>(cp->Copy());
140  nucleus_cmd->InvertDirection();
141  Add(nucleus_cmd);
142 
143  // Next, make an abstracted version of the command stream for the kernel user.
144  auto kernel_cmd = FieldPort::MakeCommandPort(cp->fletcher_schema_, cp->field_, false, kernel_cd());
145  kernel_cmd->InvertDirection();
146  kernel->Add(kernel_cmd);
147 
148  // Now, create a dirty little ACCM to solve this issue.
149  // This is dirty in the sense that the addresses are actually not streams, so there is no synchronization.
150  auto accm_inst = ArrayCmdCtrlMergerInstance(cp->name() + "_accm_inst");
151  accms.push_back(accm_inst.get());
152  // Move ownership to this component.
153  this->AddChild(std::move(accm_inst));
154  }
155  }
156 
157  // Instantiate the kernel
158  kernel_inst = AddInstanceOf(kernel.get());
159 
160  // Connect clock domain.
161  Connect(kernel_inst->port("kcd"), kcd.get());
162 
163  // Perform some magic to abstract the buffer addresses away from the ctrl stream at the kernel level.
164  // First, obtain the intended name for the kernel from the metadata of the vhdmmio component port metadata.
165  // Then, make a connection between these two components.
166  for (auto &p : vhdmmio_inst->GetAll<Port>()) {
167  if (p->meta.count(MMIO_KERNEL) > 0) {
168  auto name_intended = p->meta.at(MMIO_KERNEL);
169  auto inst_port = kernel_inst->port(name_intended);
170  if (p->dir() == Port::Dir::OUT) {
171  Connect(inst_port, p);
172  } else {
173  Connect(p, inst_port);
174  }
175  }
176  }
177 
178  // Connect all recordbatch ports
179  size_t batch_idx = 0;
180  size_t accm_idx = 0;
181  size_t buf_idx = 0;
182  for (const auto &r : recordbatches) {
183  // Connect Arrow data stream
184  for (const auto &ap : r->GetFieldPorts(FieldPort::Function::ARROW)) {
185  auto kernel_data = kernel_inst->port(ap->name());
186  auto nucleus_data = port(ap->name());
187  if (ap->dir() == Port::OUT) {
188  Connect(kernel_data, nucleus_data);
189  } else {
190  Connect(nucleus_data, kernel_data);
191  }
192  }
193 
194  // Connect unlock stream
195  for (const auto &up : r->GetFieldPorts(FieldPort::Function::UNLOCK)) {
196  auto kernel_unl = kernel_inst->port(up->name());
197  auto nucleus_unl = port(up->name());
198  Connect(kernel_unl, nucleus_unl);
199  }
200 
201  // Connect the command stream through the ACCM.
202  size_t field_idx = 0;
203  for (const auto &cp : r->GetFieldPorts(FieldPort::Function::COMMAND)) {
204  auto accm_nucleus_cmd = accms[accm_idx]->port("nucleus_cmd");
205  auto accm_kernel_cmd = accms[accm_idx]->port("kernel_cmd");
206  auto accm_ctrl = accms[accm_idx]->porta("ctrl");
207 
208  auto nucleus_cmd = port(cp->name());
209  auto kernel_cmd = kernel_inst->port(cp->name());
210 
211  Connect(nucleus_cmd, accm_nucleus_cmd);
212  Connect(accm_kernel_cmd, kernel_cmd);
213 
214  auto field_bufs = batch_desc[batch_idx].fields[field_idx].buffers;
215  for (size_t b = 0; b < field_bufs.size(); b++) {
216  auto vhdmmio_buf_addr = vhdmmio_inst->port("f_" + vhdmmio_buf_port_names[buf_idx] + "_data");
217  Connect(accm_ctrl->Append(), vhdmmio_buf_addr);
218  buf_idx++;
219  }
220  field_idx++;
221  accm_idx++;
222  }
223  batch_idx++;
224  }
225 }
