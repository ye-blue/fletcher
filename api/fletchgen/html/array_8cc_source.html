64 std::shared_ptr<Type> cmd(const std::shared_ptr<Node> &tag_width = intl(1),
65  const std::optional<std::shared_ptr<Node>> &ctrl_width) {
66  // Create fields
67  auto firstidx = RecField::Make(Vector::Make("firstIdx", 32));
68  auto lastidx = RecField::Make(Vector::Make("lastidx", 32));
69  auto tag = RecField::Make(Vector::Make("tag", tag_width));
70 
71  // Create record
72  auto cmd_record = Record::Make("command_rec", {firstidx, lastidx, tag});
73 
74  // If we want the ctrl field to be visible on this type, create that as well. This field is used to pass addresses.
75  // Depending on how advanced the developer is, we want to expose this or leave it out through the Nucleus layer.
76  if (ctrl_width) {
77  auto ctrl = RecField::Make(Vector::Make("ctrl", *ctrl_width));
78  cmd_record->AddField(ctrl, 2);
79  }
80 
81  // Create the stream type.
82  auto result = Stream::Make("command", cmd_record);
83  return result;
84 }

136 static std::shared_ptr<Component> Array(Mode mode) {
137  std::shared_ptr<BusPort> bus;
138  std::shared_ptr<Port> data;
139 
140  BusSpec spec{};
141 
142  if (mode == Mode::READ) {
143  spec.function = BusFunction::READ;
144  data = Port::Make(DataName(mode), read_data(), Port::Dir::OUT, kernel_cd());
145  bus = BusPort::Make(Port::Dir::OUT, spec);
146  } else {
147  spec.function = BusFunction::WRITE;
148  data = Port::Make(DataName(mode), write_data(), Port::Dir::IN, kernel_cd());
149  bus = BusPort::Make(Port::Dir::OUT, spec);
150  }
151 
152  std::deque<std::shared_ptr<cerata::Object>> objects;
153 
154  // Insert some parameters
155  objects.insert(objects.end(), {bus_addr_width(), bus_len_width(), bus_data_width()});
156 
157  // Insert bus strobe width for writers
158  if (mode == Mode::WRITE) {
159  objects.push_back(bus_strobe_width());
160  }
161 
162  // Insert other parameters
163  // TODO(johanpel): should be customizable and propagated from some user-defined arguments/settings
164  objects.insert(objects.end(), {
165  Parameter::Make("BUS_BURST_STEP_LEN", integer(), intl(1)),
166  Parameter::Make("BUS_BURST_MAX_LEN", integer(), intl(64)),
167  Parameter::Make("INDEX_WIDTH", integer(), intl(32)),
168  Parameter::Make("CFG", string(), strl("")),
169  Parameter::Make("CMD_TAG_ENABLE", boolean(), booll(true)),
170  Parameter::Make("CMD_TAG_WIDTH", integer(), intl(1))});
171 
172  // Insert ports
173  objects.insert(objects.end(), {
174  Port::Make("bcd", cr(), Port::Dir::IN, bus_cd()),
175  Port::Make("kcd", cr(), Port::Dir::IN, kernel_cd()),
176  bus,
177  Port::Make("cmd", cmd(intl(1), intl(1)), Port::Dir::IN, kernel_cd()),
178  Port::Make("unl", unlock(), Port::Dir::OUT, kernel_cd()),
179  data});
180 
181  auto ret = Component::Make(ArrayName(mode), objects);
182 
183  ret->SetMeta(cerata::vhdl::metakeys::PRIMITIVE, "true");
184  ret->SetMeta(cerata::vhdl::metakeys::LIBRARY, "work");
185  ret->SetMeta(cerata::vhdl::metakeys::PACKAGE, "Array_pkg");
186  return ret;
187 }

222 std::shared_ptr<Node> GetWidth(const arrow::DataType *type) {
223  switch (type->id()) {
224  // Fixed-width:
225  case arrow::Type::BOOL: return intl(1);
226  case arrow::Type::DATE32: return intl(32);
227  case arrow::Type::DATE64: return intl(64);
228  case arrow::Type::DOUBLE: return intl(64);
229  case arrow::Type::FLOAT: return intl(32);
230  case arrow::Type::HALF_FLOAT: return intl(16);
231  case arrow::Type::INT8: return intl(8);
232  case arrow::Type::INT16: return intl(16);
233  case arrow::Type::INT32: return intl(32);
234  case arrow::Type::INT64: return intl(64);
235  case arrow::Type::TIME32: return intl(32);
236  case arrow::Type::TIME64: return intl(64);
237  case arrow::Type::TIMESTAMP: return intl(64);
238  case arrow::Type::UINT8: return intl(8);
239  case arrow::Type::UINT16: return intl(16);
240  case arrow::Type::UINT32: return intl(32);
241  case arrow::Type::UINT64: return intl(64);
242 
243  // Lists:
244  case arrow::Type::LIST: return strl("OFFSET_WIDTH");
245  case arrow::Type::BINARY: return strl("OFFSET_WIDTH");
246  case arrow::Type::STRING: return strl("OFFSET_WIDTH");
247 
248  // Others:
249  default:
250  // case arrow::Type::INTERVAL: return 0;
251  // case arrow::Type::MAP: return 0;
252  // case arrow::Type::NA: return 0;
253  // case arrow::Type::DICTIONARY: return 0;
254  // case arrow::Type::UNION: return 0;
255  throw std::domain_error("Arrow type " + type->ToString() + " not supported.");
256 
257  // Structs have no width
258  case arrow::Type::STRUCT: return intl(0);
259 
260  // Other width types:
261  case arrow::Type::FIXED_SIZE_BINARY: {
262  auto t = dynamic_cast<const arrow::FixedSizeBinaryType *>(type);
263  return Literal::MakeInt(t->bit_width());
264  }
265  case arrow::Type::DECIMAL: {
266  auto t = dynamic_cast<const arrow::DecimalType *>(type);
267  return Literal::MakeInt(t->bit_width());
268  }
269  }
270 }

272 std::string GenerateConfigString(const arrow::Field &field, int level) {
273  std::string ret;
274  ConfigType ct = GetConfigType(field.type().get());
275 
276  if (field.nullable()) {
277  ret += "null(";
278  level++;
279  }
280 
281  int epc = fletcher::GetIntMeta(field, metakeys::EPC, 1);
282  int lepc = fletcher::GetIntMeta(field, metakeys::LEPC, 1);
283 
284  if (ct == ConfigType::PRIM) {
285  auto w = GetWidth(field.type().get());
286  ret += "prim(" + w->ToString();
287  level++;
288  } else if (ct == ConfigType::LISTPRIM) {
289  ret += "listprim(8";
290  level++;
291  } else if (ct == ConfigType::LIST) {
292  if (GetConfigType(field.type()->child(0)->type().get()) == ConfigType::PRIM) {
293  ret += "list";
294  } else {
295  ret += "list(";
296  }
297  level++;
298  } else if (ct == ConfigType::STRUCT) {
299  ret += "struct(";
300  level++;
301  }
302 
303  if (epc > 1 || lepc > 1) {
304  ret += ";";
305  }
306  if (epc > 1) {
307  ret += "epc=" + std::to_string(epc);
308  if (lepc > 1) {
309  ret += ",";
310  }
311  }
312  if (lepc > 1) {
313  ret += "lepc=" + std::to_string(epc);
314  }
315 
316  // Append children
317  for (int c = 0; c < field.type()->num_children(); c++) {
318  auto child = field.type()->child(c);
319  ret += GenerateConfigString(*child);
320  if (c != field.type()->num_children() - 1)
321  ret += ",";
322  }
323 
324  for (; level > 0; level--)
325  ret += ")";
326 
327  return ret;
328 }

330 std::shared_ptr<TypeMapper> GetStreamTypeMapper(Type *stream_type, Type *other) {
331  auto conversion = TypeMapper::Make(stream_type, other);
332 
333  size_t idx_stream = 0;
334  // Unused: size_t idx_record = 1;
335  size_t idx_data = 2;
336  size_t idx_dvalid = 3;
337  size_t idx_last = 4;
338 
339  auto flat_stream = conversion->flat_a();
340  for (size_t i = 0; i < flat_stream.size(); i++) {
341  auto t = flat_stream[i].type_;
342  if (t->Is(Type::STREAM)) {
343  conversion->Add(i, idx_stream);
344  } else if (t == dvalid().get()) {
345  conversion->Add(i, idx_dvalid);
346  } else if (t == last().get()) {
347  conversion->Add(i, idx_last);
348  } else if (t->Is(Type::RECORD)) {
349  // do nothing
350  } else {
351  // If it's not any of the default control signals on the stream, it must be data.
352  conversion->Add(i, idx_data);
353  }
354  }
355  return conversion;
356 }

358 std::shared_ptr<Type> GetStreamType(const arrow::Field &field, fletcher::Mode mode, int level) {
359  // The ordering of the record fields in this function determines the order in which a nested stream is type converted
360  // automatically using GetStreamTypeConverter. This corresponds to how the hardware is implemented.
361  //
362  // WARNING: Modifications to this function must be reflected in the manual hardware implementation of Fletcher
363  // components! See: hardware/arrays/ArrayConfig_pkg.vhd
364 
365  int epc = fletcher::GetIntMeta(field, metakeys::EPC, 1);
366  int lepc = fletcher::GetIntMeta(field, metakeys::LEPC, 1);
367 
368  std::shared_ptr<Type> type;
369 
370  auto arrow_id = field.type()->id();
371  const auto &name = field.name();
372 
373  switch (arrow_id) {
374  case arrow::Type::BINARY: {
375  // Special case: binary type has a length stream and byte stream. The EPC is assumed to relate to the list
376  // values, as there is no explicit child field to place this metadata in.
377 
378  std::shared_ptr<Node> e_count_width = Literal::MakeInt(static_cast<int>(ceil(log2(epc + 1))));
379  std::shared_ptr<Node> l_count_width = Literal::MakeInt(static_cast<int>(ceil(log2(lepc + 1))));
380  std::shared_ptr<Node> data_width = Literal::MakeInt(epc * 8);
381  std::shared_ptr<Node> length_width = Literal::MakeInt(lepc * 32);
382 
383  auto slave = Stream::Make(name,
384  Record::Make("slave_rec", {
385  RecField::Make("dvalid", dvalid()),
386  RecField::Make("last", last()),
387  RecField::Make("data", data(data_width)),
388  RecField::Make("count", count(e_count_width))
389  }), "", epc);
390  type = Record::Make(name + "_rec", {
391  RecField::Make("length", length(length_width)),
392  RecField::Make("count", count(l_count_width)),
393  RecField::Make("bytes", slave)
394  });
395  break;
396  }
397 
398  case arrow::Type::STRING: {
399  // Special case: string type has a length stream and utf8 character stream. The EPC is assumed to relate to the
400  // list values, as there is no explicit child field to place this metadata in.
401 
402  std::shared_ptr<Node> e_count_width = Literal::MakeInt(static_cast<int>(ceil(log2(epc + 1))));
403  std::shared_ptr<Node> l_count_width = Literal::MakeInt(static_cast<int>(ceil(log2(lepc + 1))));
404  std::shared_ptr<Node> count_width = Literal::MakeInt(static_cast<int>(ceil(log2(epc + 1))));
405  std::shared_ptr<Node> data_width = Literal::MakeInt(epc * 8);
406  std::shared_ptr<Node> length_width = Literal::MakeInt(lepc * 32);
407 
408  auto slave = Stream::Make(name,
409  Record::Make("slave_rec", {
410  RecField::Make("dvalid", dvalid()),
411  RecField::Make("last", last()),
412  RecField::Make("data", data(data_width)),
413  RecField::Make("count", count(count_width))
414  }), "", epc);
415  type = Record::Make(name + "_rec", {
416  RecField::Make("length", length(length_width)),
417  RecField::Make("count", count(l_count_width)),
418  RecField::Make("chars", slave)
419  });
420  break;
421  }
422 
423  // Lists
424  case arrow::Type::LIST: {
425  if (field.type()->num_children() != 1) {
426  throw std::runtime_error("Encountered Arrow list type with other than 1 child.");
427  }
428 
429  auto arrow_child = field.type()->child(0);
430  auto element_type = GetStreamType(*arrow_child, mode, level + 1);
431  std::shared_ptr<Node> length_width = Literal::MakeInt(32);
432 
433  auto slave = Stream::Make(name,
434  Record::Make("slave_rec", {
435  RecField::Make("dvalid", dvalid()),
436  RecField::Make("last", last()),
437  RecField::Make("data", element_type)}),
438  "", epc);
439  type = Record::Make(name + "_rec", {
440  RecField::Make("length", length(length_width)),
441  RecField::Make(arrow_child->name(), slave)
442  });
443  break;
444  }
445 
446  // Structs
447  case arrow::Type::STRUCT: {
448  if (field.type()->num_children() < 1) {
449  throw std::runtime_error("Encountered Arrow struct type without any children.");
450  }
451  std::deque<std::shared_ptr<RecField>> children;
452  for (const auto &f : field.type()->children()) {
453  auto child_type = GetStreamType(*f, mode, level + 1);
454  children.push_back(RecField::Make(f->name(), child_type));
455  }
456  type = Record::Make(name + "_rec", children);
457  break;
458  }
459 
460  // Non-nested types
461  default: {
462  type = ConvertFixedWidthType(field.type());
463  break;
464  }
465  }
466 
467  // If this is a top level field, create a stream out of it
468  if (level == 0) {
469  // Element name is empty by default.
470  std::string elements_name;
471 
472  // Create the stream record
473  auto record = Record::Make("data", {
474  RecField::Make("dvalid", dvalid()),
475  RecField::Make("last", last()),
476  RecField::Make("", type)});
477  if (field.nullable()) {
478  record->AddField(RecField::Make("validity", validity()));
479  }
480  auto stream = Stream::Make(name, record);
481  return stream;
482  } else {
483  // Otherwise just return the type
484  return type;
485  }
486 }
