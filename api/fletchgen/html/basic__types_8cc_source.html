139 std::shared_ptr<Type> ConvertFixedWidthType(const std::shared_ptr<arrow::DataType> &arrow_type) {
140  // Only need to cover fixed-width data types in this function
141  switch (arrow_type->id()) {
142  case arrow::Type::UINT8: return uint8();
143  case arrow::Type::UINT16: return uint16();
144  case arrow::Type::UINT32: return uint32();
145  case arrow::Type::UINT64: return uint64();
146  case arrow::Type::INT8: return int8();
147  case arrow::Type::INT16: return int16();
148  case arrow::Type::INT32: return int32();
149  case arrow::Type::INT64: return int64();
150  case arrow::Type::HALF_FLOAT: return float16();
151  case arrow::Type::FLOAT: return float32();
152  case arrow::Type::DOUBLE: return float64();
153  default:throw std::runtime_error("Unsupported Arrow DataType: " + arrow_type->ToString());
154  }
155 }
