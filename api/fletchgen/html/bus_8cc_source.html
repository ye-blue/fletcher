44  const std::shared_ptr<Node> &len_width,
45  const std::shared_ptr<Node> &data_width) {
46  auto raddr = RecField::Make("addr", Vector::Make("addr", addr_width));
47  auto rlen = RecField::Make("len", Vector::Make("len", len_width));
48  auto rreq_record = Record::Make("rreq:rec", {raddr, rlen});
49  auto rreq = Stream::Make("rreq:str", rreq_record);
50  auto rdata = RecField::Make(Vector::Make("data", data_width));
51  auto rlast = RecField::Make("last", last());
52  auto rdat_record = Record::Make("rdat:rec", {rdata, rlast});
53  auto rdat = Stream::Make("rdat:str", rdat_record);
54  auto result = Record::Make("BusRead", {RecField::Make("rreq", rreq), RecField::Make("rdat", rdat, true)});
55  return result;

57 std::shared_ptr<Type> bus_write(const std::shared_ptr<Node> &addr_width,
58  const std::shared_ptr<Node> &len_width,

60  auto waddr = RecField::Make(Vector::Make("addr", addr_width));
61  auto wlen = RecField::Make(Vector::Make("len", len_width));
62  auto wreq_record = Record::Make("wreq:rec", {waddr, wlen});
63  auto wreq = Stream::Make("wreq:stm", wreq_record);
64  auto wdata = RecField::Make(Vector::Make("data", data_width));
65  auto wstrobe = RecField::Make(Vector::Make("strobe", data_width / 8));
66  auto wlast = RecField::Make("last", last());
67  auto wdat_record = Record::Make("wdat:rec", {wdata, wstrobe, wlast});
68  auto wdat = Stream::Make("wdat:stm", wdat_record);
69  auto result = Record::Make("BusWrite", {RecField::Make("wreq", wreq), RecField::Make("wdat", wdat)});
70  return result;
71 }

77 static std::shared_ptr<Component> BusArbiter(BusSpec spec) {
78  // Number of slave ports
79  auto nslaves = Parameter::Make("NUM_SLAVE_PORTS", integer(), intl(0));
80 
81  // Master port
82  auto mst = BusPort::Make("mst", Port::Dir::OUT, spec);
83 
84  // Slave ports
85  auto slave_base = std::dynamic_pointer_cast<BusPort>(mst->Copy());
86  slave_base->SetName("bsv");
87  slave_base->InvertDirection();
88  auto slaves_array = PortArray::Make(slave_base, nslaves);
89 
90  std::deque<std::shared_ptr<cerata::Object>> objects;
91  objects.insert(objects.end(), {bus_addr_width(), bus_len_width(), bus_data_width()});
92  if (spec.function == BusFunction::WRITE) {
93  objects.push_back(bus_strobe_width());
94  }
95  auto empty_str = strl("");
96  objects.insert(objects.end(), {
97  nslaves,
98  Parameter::Make("ARB_METHOD", string(), strl("ROUND-ROBIN")),
99  Parameter::Make("MAX_OUTSTANDING", integer(), intl(4)),
100  Parameter::Make("RAM_CONFIG", string(), empty_str),
101  Parameter::Make("SLV_REQ_SLICES", boolean(), booll(true)),
102  Parameter::Make("MST_REQ_SLICE", boolean(), booll(true)),
103  Parameter::Make("MST_DAT_SLICE", boolean(), booll(true)),
104  Parameter::Make("SLV_DAT_SLICES", boolean(), booll(true)),
105  Port::Make("bcd", cr(), Port::Dir::IN, bus_cd()),
106  mst,
107  slaves_array,
108  });
109 
110  auto ret = Component::Make(BusArbiterName(spec), objects);
111 
112  ret->SetMeta(cerata::vhdl::metakeys::PRIMITIVE, "true");
113  ret->SetMeta(cerata::vhdl::metakeys::LIBRARY, "work");
114  ret->SetMeta(cerata::vhdl::metakeys::PACKAGE, "Interconnect_pkg");
115  return ret;
116 }

156 std::shared_ptr<Type> bus(BusSpec spec) {
157  std::shared_ptr<Type> result;
158  // Check if the bus type already exists in the type pool
159  auto bus_typename = spec.ToBusTypeName();
160  auto optional_existing_bus_type = cerata::default_type_pool()->Get(bus_typename);
161  if (optional_existing_bus_type) {
162  FLETCHER_LOG(DEBUG, "BUS type already exists in default pool.");
163  result = optional_existing_bus_type.value()->shared_from_this();
164  } else {
165  auto addr_width = intl(spec.addr_width);
166  auto len_width = intl(spec.len_width);
167  auto data_width = intl(spec.data_width);
168  switch (spec.function) {
169  case BusFunction::READ:result = bus_read(addr_width, len_width, data_width);
170  break;
171  case BusFunction::WRITE:result = bus_write(addr_width, len_width, data_width);
172  break;
173  }
174  }
175  return result;
176 }
