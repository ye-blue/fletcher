45 std::shared_ptr<Type> mmio(MmioSpec spec) {
46  auto mmio_typename = spec.ToMMIOTypeName();
47  auto optional_existing_mmio_type = cerata::default_type_pool()->Get(mmio_typename);
48  if (optional_existing_mmio_type) {
49  FLETCHER_LOG(DEBUG, "MMIO Type already exists in default pool.");
50  return optional_existing_mmio_type.value()->shared_from_this();
51  } else {
52  auto new_mmio_type = Record::Make(mmio_typename, {
53  NoSep(RecField::Make("aw", Stream::Make(Record::Make("aw", {
54  RecField::Make("addr", Vector::Make(spec.addr_width))})))),
55  NoSep(RecField::Make("w", Stream::Make(Record::Make("w", {
56  RecField::Make("data", Vector::Make(spec.data_width)),
57  RecField::Make("strb", Vector::Make(spec.data_width / 8))
58  })))),
59  NoSep(RecField::Make("b", Stream::Make(Record::Make("b", {
60  RecField::Make("resp", Vector::Make(2))})), true)),
61  NoSep(RecField::Make("ar", Stream::Make(Record::Make("ar", {
62  RecField::Make("addr", Vector::Make(spec.addr_width))
63  })))),
64  NoSep(RecField::Make("r", Stream::Make(Record::Make("r", {
65  RecField::Make("data", Vector::Make(spec.data_width)),
66  RecField::Make("resp", Vector::Make(2))})), true)),
67  });
68  cerata::default_type_pool()->Add(new_mmio_type);
69  return new_mmio_type;
70  }
71 }

98 void GenerateVhdmmioYaml(const std::vector<fletcher::RecordBatchDescription> &batches,
99  const std::vector<MmioReg> &custom_regs) {
100  auto ofs = std::ofstream("fletchgen.mmio.yaml");
101  ofs << "metadata:\n"
102  " name: mmio\n"
103  " doc: Fletchgen generated MMIO configuration.\n"
104  " \n"
105  "entity:\n"
106  " bus-flatten: yes\n"
107  " bus-prefix: mmio_\n"
108  " clock-name: kcd_clk\n"
109  " reset-name: kcd_reset\n"
110  "\n"
111  "features:\n"
112  " bus-width: 32\n"
113  " optimize: yes\n"
114  "\n"
115  "interface:\n"
116  " flatten: yes\n"
117  "\n"
118  "fields: \n"
119  " - address: 0\n"
120  " register-name: control\n"
121  " bitrange: 0\n"
122  " name: start\n"
123  " behavior: strobe\n"
124  "\n"
125  " - address: 0\n"
126  " register-name: control\n"
127  " bitrange: 1\n"
128  " name: stop\n"
129  " behavior: strobe\n"
130  "\n"
131  " - address: 0\n"
132  " register-name: control\n"
133  " bitrange: 2\n"
134  " name: reset\n"
135  " behavior: strobe\n"
136  "\n"
137  " - address: 4\n"
138  " register-name: status\n"
139  " bitrange: 0\n"
140  " name: idle\n"
141  " behavior: status\n"
142  "\n"
143  " - address: 4\n"
144  " register-name: status\n"
145  " bitrange: 1\n"
146  " name: busy\n"
147  " behavior: status\n"
148  "\n"
149  " - address: 4\n"
150  " register-name: status\n"
151  " bitrange: 2\n"
152  " name: done\n"
153  " behavior: status\n"
154  "\n"
155  " - address: 8\n"
156  " bitrange: 63..0\n"
157  " name: result\n"
158  " behavior: status\n"
159  "\n";
160 
161  int addr = 4 * FLETCHER_REG_SCHEMA;
162  for (const auto &r : batches) {
163  ofs << " - address: " << addr << "\n";
164  ofs << " doc: " << "First index to process from Arrow RecordBatch " << r.name << "\n";
165  ofs << " name: " << r.name << "_firstidx" << "\n";
166  ofs << " behavior: control\n";
167  addr += 4;
168  ofs << " - address: " << addr << "\n";
169  ofs << " doc: " << "Last index to process from Arrow RecordBatch " << r.name << "\n";
170  ofs << " name: " << r.name << "_lastidx" << "\n";
171  ofs << " behavior: control\n";
172  addr += 4;
173  ofs << "\n";
174  }
175 
176  for (const auto &r : batches) {
177  for (const auto &f : r.fields) {
178  for (const auto &b : f.buffers) {
179  ofs << " - address: " << addr << "\n";
180  ofs << " doc: " << "RecordBatch " + r.name + " field " + b.desc_[0] + "\n";
181  ofs << " name: " << r.name + "_" + fletcher::ToString(b.desc_) << "\n";
182  ofs << " bitrange: 63..0\n";
183  ofs << " behavior: control\n";
184  ofs << "\n";
185  addr += 8;
186  }
187  }
188  }
189 
190  for (const auto &r : custom_regs) {
191  // Let the user know where this turned out to be.
192  FLETCHER_LOG(INFO, "Custom register: \"" + r.name + "\" mapped to address offset " + std::to_string(addr));
193  // Write the custom register yaml lines
194  ofs << " - address: " << addr << "\n";
195  ofs << " doc: " << "Custom register " + r.name + "\n";
196  ofs << " name: " << r.name << "\n";
197  ofs << " bitrange: " << r.width - 1 << "..0\n";
198  ofs << " behavior: " << ToString(r.behavior) << "\n";
199  ofs << "\n";
200  addr += r.addr_space_used;
201  }
202 }

230 std::shared_ptr<Component> GenerateMmioComponent(const std::vector<fletcher::RecordBatchDescription> &batches,
231  const std::vector<MmioReg> &custom_regs,
232  std::vector<std::string> *buf_port_names) {
233  auto reg32 = Vector::Make(32);
234  auto reg64 = Vector::Make(64);
235 
236  auto kcd = Port::Make("kcd", cr(), Port::Dir::IN, kernel_cd());
237 
238  // Kernel ports, i.e. ports that should be exposed to the kernel
239  auto comp = Component::Make("mmio",
240  {kcd,
241  MmioKernelPort("start", bit(), Port::Dir::OUT),
242  MmioKernelPort("stop", bit(), Port::Dir::OUT),
243  MmioKernelPort("reset", bit(), Port::Dir::OUT),
244  MmioKernelPort("idle", bit()),
245  MmioKernelPort("busy", bit()),
246  MmioKernelPort("done", bit()),
247  MmioKernelPort("result", reg64)});
248 
249  for (const auto &reg : custom_regs) {
250  auto kp = MmioKernelPort(reg.name,
251  cerata::Vector::Make(reg.width),
252  reg.behavior == MmioReg::Behavior::CONTROL ? Port::Dir::OUT : Port::Dir::IN);
253  comp->Add(kp);
254  }
255 
256  for (const auto &b : batches) {
257  comp->Add(MmioKernelPort(b.name + "_firstidx", reg32, Port::Dir::OUT));
258  comp->Add(MmioKernelPort(b.name + "_lastidx", reg32, Port::Dir::OUT));
259  }
260 
261  for (const auto &r : batches) {
262  for (const auto &f : r.fields) {
263  for (const auto &b : f.buffers) {
264  auto buffer_port_name = r.name + "_" + fletcher::ToString(b.desc_);
265  comp->Add(MmioBufferPort(buffer_port_name, reg64, Port::Dir::OUT));
266  buf_port_names->push_back(buffer_port_name);
267  }
268  }
269  }
270 
271  auto bus = Port::Make("mmio", mmio(), Port::Dir::IN, kernel_cd());
272 
273  comp->Add(bus);
274 
275  // This will be a primitive component generated by VHDMMIO
276  comp->SetMeta(cerata::vhdl::metakeys::PRIMITIVE, "true");
277  comp->SetMeta(cerata::vhdl::metakeys::LIBRARY, "work");
278  comp->SetMeta(cerata::vhdl::metakeys::PACKAGE, "mmio_pkg");
279 
280  return comp;
281 }
