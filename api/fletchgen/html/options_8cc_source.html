22 bool Options::Parse(Options *options, int argc, char **argv) {
23  CLI::App app{"Fletchgen - The Fletcher Design Generator"};
24 
25  app.get_formatter()->column_width(34);
26 
27  // Required options:
28  app.add_option("-i,--input", options->schema_paths,
29  "List of files with Arrow Schemas to base design on."
30  "Example: --input file1.fbs file2.fbs file3.fbs")
31  ->check(CLI::ExistingFile);
32 
33  // Naming options:
34  app.add_option("-n,--kernel_name", options->kernel_name,
35  "Name of the accelerator kernel.");
36 
37  // Simulation options:
38  app.add_option("-r,--recordbatch_input", options->recordbatch_paths,
39  "List of files with Arrow RecordBatches to base design on and use in simulation memory models."
40  "Schemas contained in these RecordBatches may be skipped for the --input option.");
41  app.add_option("-s,--recordbatch_output", options->srec_out_path,
42  "Memory model contents output file (formatted as SREC).");
43  app.add_option("-t,--srec_dump", options->srec_sim_dump,
44  "Path to dump memory model contents to after simulation (formatted as SREC).");
45 
46  // Output options:
47  app.add_option("-o,--output_path", options->output_dir,
48  "Path to the output directory to place the generated files. (Default: . )");
49 
50  app.add_option("-l,--language", options->languages,
51  "Select the output languages for your design. Each type of output will be stored in a "
52  "seperate subfolder (e.g. <output folder>/vhdl/...). \n"
53  "Available languages:\n"
54  " vhdl: Export as VHDL files (default).\n"
55  " dot : Export as DOT graphs.");
56 
57  app.add_flag("-f,--force", options->overwrite,
58  "Force overwriting source code files if they exists already. If this flag is *not* used and the source "
59  "file exists already in the specified path, the output filename will be <filename>.<extension>t. This "
60  "file IS always overwritten. This applies only to files that the user should modify.");
61 
62  app.add_option("--regs", options->regs,
63  "Names of custom registers in the following format: \"<behavior>:<width>:<name>\", where <behavior> "
64  "is one character from the following options:\n"
65  " c : (control) register content is controlled by host-side software.\n"
66  " s : (status) register content is controlled by hardware kernel.\n"
67  "Example: \"-reg32 c:my_host_to_kernel_signaling_reg s:my_kernel_to_host_signaling_reg\"");
68 
69  app.add_flag("--axi", options->axi_top,
70  "Generate AXI top-level template (VHDL only).");
71  app.add_flag("--sim", options->sim_top,
72  "Generate simulation top-level template (VHDL only).");
73  app.add_flag("--vivado_hls", options->vivado_hls,
74  "Generate a Vivado HLS kernel template.");
75 
76  // Other options:
77  // TODO(johanpel): implement the quiet and verbose options
78  /*
79  app.add_flag("-q,--quiet", options->quiet,
80  "Surpress all stdout.");
81  app.add_flag("-v,--verbose", options->verbose,
82  "More detailed information on stdout.");
83  */
84 
85  try {
86  app.parse(argc, argv);
87  } catch (CLI::CallForHelp &e) {
88  std::cout << app.help();
89  options->quit = true;
90  return true;
91  } catch (CLI::Error &e) {
92  FLETCHER_LOG(ERROR, e.get_name() + ":\n" + e.what());
93  return false;
94  }
95 
96  // Load input files
97  if (!options->LoadRecordBatches()) return false;
98  if (!options->LoadSchemas()) return false;
99 
100  return true;
101 }

135 bool Options::LoadRecordBatches() {
136  for (const auto &path : recordbatch_paths) {
137  std::vector<std::shared_ptr<arrow::RecordBatch>> rbs;
138  FLETCHER_LOG(INFO, "Loading RecordBatch(es) from " + path);
139  if (!fletcher::ReadRecordBatchesFromFile(path, &rbs)) {
140  return false;
141  }
142  recordbatches.insert(recordbatches.end(), rbs.begin(), rbs.end());
143  }
144  return true;
145 }
146 
147 bool Options::LoadSchemas() {
148  for (const auto &path : schema_paths) {
149  std::shared_ptr<arrow::Schema> schema;
150  FLETCHER_LOG(INFO, "Loading Schema from " + path);
151  if (!fletcher::ReadSchemaFromFile(path, &schema)) {
152  return false;
153  }
154  schemas.push_back(schema);
155  }
156  return true;
157 }
