40 Mantle::Mantle(std::string name,
41  SchemaSet schema_set,
42  const std::vector<fletcher::RecordBatchDescription> &batch_desc,
43  const std::vector<MmioReg> &custom_regs)
44  : Component(std::move(name)), schema_set_(std::move(schema_set)) {
45 
46  // Add default ports
47  auto bcr = Port::Make("bcd", cr(), Port::Dir::IN, bus_cd());
48  auto kcr = Port::Make("kcd", cr(), Port::Dir::IN, kernel_cd());
49  auto regs = MmioPort::Make(Port::Dir::IN);
50  Add({bcr, kcr, regs});
51 
52  Add(bus_addr_width());
53 
54  // Create and add every RecordBatch/Writer.
55  for (const auto &fs : schema_set_.schemas()) {
56  auto rb = RecordBatch::Make(fs);
57  recordbatch_components_.push_back(rb);
58  auto rb_inst = AddInstanceOf(rb.get());
59  recordbatch_instances_.push_back(rb_inst);
60  rb_inst->port("kcd") <<= kcr;
61  rb_inst->port("bcd") <<= bcr;
62  }
63 
64  // Create and add the Nucleus.
65  nucleus_ = Nucleus::Make(schema_set_.name(),
66  cerata::ToRawPointers(recordbatch_components()),
67  batch_desc,
68  custom_regs);
69  nucleus_inst_ = AddInstanceOf(nucleus_.get());
70  nucleus_inst_->port("kcd") <<= kcr;
71  nucleus_inst_->port("mmio") <<= regs;
72 
73  // Connect all Arrow field derived ports.
74  for (const auto &r : recordbatch_instances_) {
75  auto field_ports = r->GetAll<FieldPort>();
76  for (const auto &fp : field_ports) {
77  if (fp->function_ == FieldPort::Function::ARROW) {
78  // If the port is an output, it's an input for the kernel and vice versa.
79  // Connect the ports and remember the edge.
80  std::shared_ptr<cerata::Edge> e;
81  if (fp->dir() == cerata::Term::Dir::OUT) {
82  e = Connect(nucleus_inst_->port(fp->name()), fp);
83  } else {
84  e = Connect(fp, nucleus_inst_->port(fp->name()));
85  }
86  } else if (fp->function_ == FieldPort::Function::COMMAND) {
87  Connect(fp, nucleus_inst_->port(fp->name()));
88  } else if (fp->function_ == FieldPort::Function::UNLOCK) {
89  Connect(nucleus_inst_->port(fp->name()), fp);
90  }
91  }
92  }
93 
94  std::deque<BusSpec> bus_specs;
95  std::deque<BusPort *> bus_ports;
96 
97  // For all the bus interfaces, figure out which unique bus specifications there are.
98  for (const auto &r : recordbatch_instances_) {
99  auto r_bus_ports = r->GetAll<BusPort>();
100  for (const auto &b : r_bus_ports) {
101  bus_specs.push_back(b->spec_);
102  bus_ports.push_back(b);
103  }
104  }
105 
106  // Leave only unique bus specs.
107  auto last = std::unique(bus_specs.begin(), bus_specs.end());
108  bus_specs.erase(last, bus_specs.end());
109 
110  // Generate a BusArbiterVec for every unique bus specification.
111  for (const auto &spec : bus_specs) {
112  FLETCHER_LOG(DEBUG, "Adding bus arbiter for: " + spec.ToString());
113  auto arbiter_instance = BusArbiterInstance(spec);
114  auto arbiter = arbiter_instance.get();
115  AddChild(std::move(arbiter_instance));
116  arbiter->par(bus_addr_width()->name()) <<= intl(spec.addr_width);
117  arbiter->par(bus_data_width()->name()) <<= intl(spec.data_width);
118  arbiter->par(bus_len_width()->name()) <<= intl(spec.len_width);
119  if (spec.function == BusFunction::WRITE) {
120  arbiter->par(bus_strobe_width()->name()) <<= intl(static_cast<int>(spec.data_width / 8));
121  }
122  arbiters_[spec] = arbiter;
123  // Create the bus port on the mantle level.
124  auto master = BusPort::Make(ArbiterMasterName(spec), Port::Dir::OUT, spec);
125  Add(master);
126  // TODO(johanpel): actually support multiple bus specs
127  // Connect the arbiter master port to the mantle master port.
128  master <<= arbiter->port("mst");
129  // Connect the bus clock domain.
130  arbiter->port("bcd") <<= bcr;
131  }
132 
133  // Connect bus ports to the arbiters.
134  for (const auto &bp : bus_ports) {
135  // Get the arbiter port.
136  auto arbiter = arbiters_.at(bp->spec_);
137  auto arbiter_port_array = arbiter->porta("bsv");
138  // Generate a mapper. TODO(johanpel): implement bus ports with same spec to map automatically.
139  auto mapper = TypeMapper::MakeImplicit(bp->type(), arbiter_port_array->type());
140  bp->type()->AddMapper(mapper);
141  Connect(arbiter_port_array->Append(), bp);
142  }
143 }
