108 template<typename T>
109 class MappingMatrix {
110  public:
116  MappingMatrix(int64_t height, int64_t width) : height_(height), width_(width) {
117  elements_ = std::vector<T>(height_ * width_, static_cast<T>(0));
118  }
119 
121  static MappingMatrix Identity(int64_t dim) {
122  MappingMatrix ret(dim, dim);
123  for (int64_t i = 0; i < dim; i++) {
124  ret(i, i) = 1;
125  }
126  }
127 
129  int64_t height() { return height_; }
131  int64_t width() { return width_; }
132 
134  T &get(int64_t y, int64_t x) {
135  if (y >= height_ || x >= width_) {
136  CERATA_LOG(FATAL, "Indices exceed matrix dimensions.");
137  }
138  return elements_[width_ * y + x];
139  }
140 
142  [[nodiscard]] const T &get(int64_t y, int64_t x) const {
143  if (y >= height_ || x >= width_) {
144  CERATA_LOG(FATAL, "Indices exceed matrix dimensions.");
145  }
146  return elements_[width_ * y + x];
147  }
148 
150  T &operator()(int64_t y, int64_t x) {
151  return get(y, x);
152  }
153 
155  const T &operator()(int64_t y, int64_t x) const {
156  return get(y, x);
157  }
158 
160  [[nodiscard]] T MaxOfColumn(int64_t x) const {
161  T max = 0;
162  for (int64_t y = 0; y < height_; y++) {
163  if (get(y, x) > max) {
164  max = get(y, x);
165  }
166  }
167  return max;
168  }
169 
171  [[nodiscard]] T MaxOfRow(int64_t y) const {
172  T max = 0;
173  for (int64_t x = 0; x < width_; x++) {
174  if (get(y, x) > max) {
175  max = get(y, x);
176  }
177  }
178  return max;
179  }
180 
182  std::deque<std::pair<int64_t, T>> mapping_column(int64_t x) {
183  std::deque<std::pair<int64_t, T>> ret;
184  for (int64_t y = 0; y < height_; y++) {
185  auto val = get(y, x);
186  if (val > 0) {
187  ret.emplace_back(y, val);
188  }
189  }
190  std::sort(ret.begin(), ret.end(), [](const auto &x, const auto &y) -> bool {
191  return x.second < y.second;
192  });
193  return ret;
194  }
195 
197  std::deque<std::pair<int64_t, T>> mapping_row(int64_t y) {
198  using pair = std::pair<int64_t, T>;
199  std::deque<pair> ret;
200  for (int64_t x = 0; x < width_; x++) {
201  auto val = get(y, x);
202  if (val > 0) {
203  ret.emplace_back(x, val);
204  }
205  }
206  std::sort(ret.begin(), ret.end(), [](const auto &x, const auto &y) -> bool {
207  return x.second < y.second;
208  });
209  return ret;
210  }
211 
213  MappingMatrix &SetNext(int64_t y, int64_t x) {
214  get(y, x) = std::max(MaxOfColumn(x), MaxOfRow(y)) + 1;
215  return *this;
216  }
217 
219  [[nodiscard]] MappingMatrix Transpose() const {
220  MappingMatrix ret(width_, height_);
221  for (int64_t y = 0; y < height_; y++) {
222  for (int64_t x = 0; x < width_; x++) {
223  ret(x, y) = get(y, x);
224  }
225  }
226  return ret;
227  }
228 
230  [[nodiscard]] MappingMatrix Empty() const {
231  MappingMatrix ret(height_, width_);
232  return ret;
233  }
234 
236  std::string ToString() {
237  std::stringstream ret;
238  for (int64_t y = 0; y < height_; y++) {
239  for (int64_t x = 0; x < width_; x++) {
240  ret << std::setw(3) << std::right << std::to_string(get(y, x)) << " ";
241  }
242  ret << "\n";
243  }
244  return ret.str();
245  }
246 
247  private:
249  std::vector<T> elements_;
251  int64_t height_;
253  int64_t width_;
254 };
