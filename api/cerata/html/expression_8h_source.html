108 inline std::shared_ptr<Node> operator SYMBOL (const std::shared_ptr<const Node>& lhs, int rhs) { \
109  if (lhs->IsLiteral()) { \
110  auto li = std::dynamic_pointer_cast<const Literal>(lhs); \

112  return default_node_pool()->GetLiteral(li->IntValue() SYMBOL rhs); \
113  } \
114  } \

116 } \
