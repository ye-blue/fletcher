30 static std::string GenerateTypeDecl(const Type &type,
31  const std::optional<std::shared_ptr<Node>> &multiplier = std::nullopt) {
32  switch (type.id()) {
33  default: {
34  if (!multiplier) {
35  return "std_logic";
36  } else {
37  return "std_logic_vector(" + (*multiplier - 1)->ToString() + " downto 0)";
38  }
39  }
40  case Type::VECTOR: {
41  auto &vec = dynamic_cast<const Vector &>(type);
42  auto width = vec.width();
43  if (width) {
44  auto wnode = width.value();
45  if (!multiplier) {
46  auto expr = *wnode - 1;
47  return "std_logic_vector(" + expr->ToString() + " downto 0)";
48  } else {
49  auto expr = *multiplier * wnode - 1;
50  return "std_logic_vector(" + expr->ToString() + " downto 0)";
51  }
52  } else {
53  return "<incomplete type>";
54  }
55  }
56  case Type::RECORD: {
57  auto record = dynamic_cast<const Record &>(type);
58  return record.name();
59  }
60  case Type::INTEGER: {
61  return "integer";
62  }
63  case Type::NATURAL: {
64  return "natural";
65  }
66  case Type::STREAM: {
67  auto stream = dynamic_cast<const Stream &>(type);
68  return GenerateTypeDecl(*stream.element_type());
69  }
70  case Type::STRING: {
71  return "string";
72  }
73  case Type::BOOLEAN: {
74  return "boolean";
75  }
76  }
77 }

79 Block Decl::Generate(const Parameter &par, int depth) {
80  Block ret(depth);
81  Line l;
82  l << to_upper(par.name()) << " : " << GenerateTypeDecl(*par.type());
83  if (par.GetValue()) {
84  Node *val = par.GetValue().value();
85  l << " := " << val->ToString();
86  }
87  ret << l;
88  return ret;

91 Block Decl::Generate(const Port &port, int depth) {
92  Block ret(depth);
93  // Filter out abstract types and flatten
94  auto flat_types = FilterForVHDL(Flatten(port.type()));
95  for (const auto &ft : flat_types) {
96  Line l;
97  auto port_name_prefix = port.name();
98  l << ft.name(NamePart(port_name_prefix, true)) << " : ";
99  if (ft.invert_) {
100  l << ToString(Term::Invert(port.dir())) + " ";
101  } else {
102  l << ToString(port.dir()) + " ";
103  }
104  l << GenerateTypeDecl(*ft.type_);
105  ret << l;
106  }
107  return ret;
108 }

114

161 MultiBlock Decl::Generate(const Component &comp, bool entity) {
162  MultiBlock ret(1);
163 
164  if (entity) {
165  ret.indent = 0;
166  }
167 
168  // Header
169  Block h(ret.indent), f(ret.indent);
170  Line hl, fl;
171  if (entity) {
172  hl << "entity " + comp.name() + " is";
173  } else {
174  hl << "component " + comp.name() + " is";
175  }
176  h << hl;
177  ret << h;
178 
179  // Generics
180  std::deque<Parameter *> parameters = comp.GetAll<Parameter>();
181  if (!parameters.empty()) {
182  Block gdh(ret.indent + 1);
183  Block gd(ret.indent + 2);
184  Block gdf(ret.indent + 1);
185  Line gh, gf;
186  gh << "generic (";
187  gdh << gh;
188  for (Parameter *gen : parameters) {
189  auto g = Decl::Generate(*gen, ret.indent + 2);
190  if (gen != parameters.back()) {
191  g << ";";
192  } else {
193  g <<= ";";
194  }
195  gd << g;
196  }
197  gf << ");";
198  gdf << gf;
199  ret << gdh << gd << gdf;
200  }
201 
202  auto ports = comp.GetAll<Port>();
203  auto array_ports = comp.GetAll<PortArray>();
204  if (!ports.empty() || !array_ports.empty()) {
205  Block pdh(ret.indent + 1);
206  Block pd(ret.indent + 2);
207  Block pdf(ret.indent + 1);
208  Line ph, pf;
209  ph << "port (";
210  pdh << ph;
211  // Process ports
212  for (const auto &port : ports) {
213  auto g = Decl::Generate(*port, ret.indent + 2);
214  if (port != ports.back() || !array_ports.empty()) {
215  g << ";";
216  } else {
217  g <<= ";";
218  }
219  pd << g;
220  }
221  // Process array ports
222  for (const auto &array_port : array_ports) {
223  auto g = Decl::Generate(*array_port, ret.indent + 2);
224  if (array_port != array_ports.back()) {
225  g << ";";
226  } else {
227  g <<= ";";
228  }
229  pd << g;
230  }
231  pf << ");";
232  pdf << pf;
233  ret << pdh << pd << pdf;
234  }
235 
236  if (entity) {
237  fl << "end entity;";
238  } else {
239  fl << "end component;";
240  }
241 
242  f << fl;
243 
244  ret << f;
245 
246  return ret;
247 }
