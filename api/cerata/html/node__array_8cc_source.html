32 static std::shared_ptr<Node> IncrementNode(const Node &node) {
33  if (node.IsLiteral() || node.IsExpression()) {
34  return node + 1;
35  } else if (node.IsParameter()) {
36  // If the node is a parameter.
37  auto &param = dynamic_cast<const Parameter &>(node);
38  // Make a copy of the parameter.
39  std::shared_ptr<Node> new_param = std::dynamic_pointer_cast<Node>(param.Copy());
40  // Figure out who should own the copy.
41  if (param.parent()) {
42  param.parent().value()->Add(new_param);
43  } else {
44  // If theres no parent, place it in the default node pool.
45  // TODO(johanpel): this is a bit ugly.
46  default_node_pool()->Add(new_param);
47  }
48  if (param.GetValue()) {
49  auto param_value = param.GetValue().value();
50  // Recurse until we reach the literal.
51  new_param <<= IncrementNode(*param_value);
52  } else {
53  // Otherwise connect an integer literal of 1 to the parameter.
54  new_param <<= intl(1);
55  }
56  return new_param;
57  } else {
58  CERATA_LOG(FATAL, "Cannot increment node " + node.name() + " of type " + ToString(node.node_id()));
59  }
60 }

70 void NodeArray::IncrementSize() {
71  if (size_ != nullptr) {
72  if (parent_) {
73  parent_.value()->Remove(size_.get());
74  }
75  auto new_size = IncrementNode(*size_);
76  SetSize(new_size);
77  } else {
78  throw std::runtime_error("Corrupted NodeArray. Size is nullptr.");
79  }
80 }

82 Node *NodeArray::Append(bool increment_size) {
83  auto elem = std::dynamic_pointer_cast<Node>(base_->Copy());
84  if (parent()) {
85  elem->SetParent(*parent());
86  }
87  elem->SetArray(this);
88  nodes_.push_back(elem);
89  if (increment_size) {
90  IncrementSize();
91  }
92  return elem.get();
93 }
