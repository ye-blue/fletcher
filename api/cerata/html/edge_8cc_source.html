42 static void CheckDomains(Node *src, Node *dst) {
43  if ((src->IsPort() || src->IsSignal()) && (dst->IsPort() || dst->IsSignal())) {
44  auto src_dom = dynamic_cast<Synchronous *>(src)->domain();
45  auto dst_dom = dynamic_cast<Synchronous *>(dst)->domain();
46  if (src_dom != dst_dom) {
47  std::stringstream warning;
48  warning << "Attempting to connect Synchronous nodes, but clock domains differ.\n";
49 
50  warning << "Src: [" + src->ToString() + "] in domain: [" + dst_dom->name() + "]";
51  if (src->parent()) {
52  warning << " on parent: [" + src->parent().value()->name() + "]";
53  }
54 
55  warning << "\nDst: [" + dst->ToString() + "] in domain: [" + src_dom->name() + "]";
56  if (dst->parent()) {
57  warning << " on parent: [" + dst->parent().value()->name() + "]";
58  }
59 
60  warning << "\nAutomated CDC crossings are not yet implemented or instantiated.";
61  warning << "This behavior may cause incorrect designs.";
62  CERATA_LOG(WARNING, warning.str());
63  }
64  }
65 }

67 std::shared_ptr<Edge> Connect(Node *dst, Node *src) {
68  // Check for potential errors
69  if (src == nullptr) {
70  CERATA_LOG(ERROR, "Source node is null");
71  return nullptr;
72  } else if (dst == nullptr) {
73  CERATA_LOG(ERROR, "Destination node is null");
74  return nullptr;
75  }
76 
77  // Check if the clock domains correspond. Currently, this doesn't result in an error as automated CDC support is not
78  // in place yet. Just generate a warning for now:
79  CheckDomains(src, dst);
80 
81  // Check if the types can be mapped onto each other
82  if (!src->type()->GetMapper(dst->type())) {
83  CERATA_LOG(FATAL, "No known type mapping available for connection between node "
84  + dst->ToString() + " (" + dst->type()->ToString() + ") and ("
85  + src->ToString() + " (" + src->type()->ToString() + ")");
86  }
87 
88  // If the destination is a terminator
89  if (dst->IsPort()) {
90  auto port = dynamic_cast<Port *>(dst);
91  // Check if it has a parent
92  if (dst->parent()) {
93  auto parent = *dst->parent();
94  if (parent->IsInstance() && port->IsOutput()) {
95  // If the parent is an instance, and the terminator node is an output, then we may not drive it.
96  CERATA_LOG(FATAL, "Cannot drive instance port " + dst->ToString() + " of mode output.");
97  } else if (parent->IsComponent() && port->IsInput()) {
98  // If the parent is a component, and the terminator node is an input, then we may not drive it.
99  CERATA_LOG(FATAL, "Cannot drive component port " + dst->ToString() + " of mode input.");
100  }
101  }
102  }
103 
104  // If the source is a terminator
105  if (src->IsPort()) {
106  auto port = dynamic_cast<Port *>(src);
107  // Check if it has a parent
108  if (src->parent()) {
109  auto parent = *src->parent();
110  if (parent->IsInstance() && port->IsInput()) {
111  // If the parent is an instance, and the terminator node is an input, then we may not source from it.
112  CERATA_LOG(FATAL, "Cannot source from instance port " + src->ToString() + " of mode input "
113  "on " + parent->ToString());
114  } else if (parent->IsComponent() && port->IsOutput()) {
115  // If the parent is a component, and the terminator node is an output, then we may not source from it.
116  CERATA_LOG(FATAL, "Cannot source from component port " + src->ToString() + " of mode output "
117  "on " + parent->ToString());
118  }
119  }
120  }
121 
122  std::string edge_name = src->name() + "_to_" + dst->name();
123  auto edge = Edge::Make(edge_name, dst, src);
124  src->AddEdge(edge);
125  dst->AddEdge(edge);
126  return edge;
127 }

149 std::deque<Edge *> GetAllEdges(const Graph &graph) {
150  std::deque<Edge *> all_edges;
151 
152  // Get all normal nodes
153  for (const auto &node : graph.GetAll<Node>()) {
154  auto out_edges = node->sinks();
155  for (const auto &e : out_edges) {
156  all_edges.push_back(e);
157  }
158  auto in_edges = node->sources();
159  for (const auto &e : in_edges) {
160  all_edges.push_back(e);
161  }
162  }
163 
164  for (const auto &array : graph.GetAll<NodeArray>()) {
165  for (const auto &node : array->nodes()) {
166  auto out_edges = node->sinks();
167  for (const auto &e : out_edges) {
168  all_edges.push_back(e);
169  }
170  auto in_edges = node->sources();
171  for (const auto &e : in_edges) {
172  all_edges.push_back(e);
173  }
174  }
175  }
176 
177  if (graph.IsComponent()) {
178  auto &comp = dynamic_cast<const Component &>(graph);
179  for (const auto &g : comp.children()) {
180  auto child_edges = GetAllEdges(*g);
181  all_edges.insert(all_edges.end(), child_edges.begin(), child_edges.end());
182  }
183  }
184 
185  return all_edges;
186 }

188 std::shared_ptr<Signal> insert(Edge *edge, const std::string &name_prefix, std::optional<Graph *> new_owner) {
189  auto src = edge->src();
190  auto dst = edge->dst();
191 
192  // Make sure we're inserting between signal/port nodes.
193  if (!(src->IsPort() || src->IsSignal()) && (dst->IsPort() || dst->IsSignal())) {
194  CERATA_LOG(FATAL, "Attempting to insert signal node on edge between non-port/signal node.");
195  }
196 
197  // When they were connected, their clock domains must have matched. Just grab the domain from the source.
198  std::shared_ptr<ClockDomain> domain;
199  if (src->IsPort()) domain = src->AsPort().domain();
200  if (src->IsSignal()) domain = src->AsSignal().domain();
201 
202  // Get the destination type
203  auto type = src->type();
204  auto name = name_prefix + src->name();
205  // Create the signal and take shared ownership of the type
206  auto signal = Signal::Make(name, type->shared_from_this(), domain);
207 
208  // Share ownership of the new signal with the potential new_owner
209  if (new_owner) {
210  (*new_owner)->Add(signal);
211  }
212 
213  // Remove the original edge from the source and destination node
214  src->RemoveEdge(edge);
215  dst->RemoveEdge(edge);
216  // From this moment onward, the edge may be deconstructed and should not be used anymore.
217  // Make the new connections, effectively creating two new edges on either side of the signal node.
218  signal <<= src;
219  dst <<= signal;
220  // Return the new signal
221  return signal;
222 }
