47 std::string Type::ToString(bool show_meta, bool show_mappers) const {
48  std::string ret;
49  switch (id_) {
50  case CLOCK : ret = name() + ":Clk";
51  break;
52  case RESET : ret = name() + ":Rec";
53  break;
54  case BIT : ret = name() + ":Bit";
55  break;
56  case VECTOR : ret = name() + ":Vec";
57  break;
58  case INTEGER: ret = name() + ":Int";
59  break;
60  case STRING : ret = name() + ":Str";
61  break;
62  case BOOLEAN: ret = name() + ":Bo";
63  break;
64  case RECORD : ret = name() + ":Rec";
65  break;
66  case STREAM : ret = name() + ":Stm";
67  break;
68  default :throw std::runtime_error("Corrupted Type ID.");
69  }
70 
71  if (show_meta || show_mappers) {
72  ret += "[";
73  // Append metadata
74  ret += ::cerata::ToString(meta);
75  if (show_mappers && !mappers_.empty()) {
76  ret += " ";
77  }
78 
79  // Append mappers
80  if (show_mappers && !mappers_.empty()) {
81  ret += "mappers={";
82  size_t i = 0;
83  for (const auto &m : mappers_) {
84  ret += m->b()->ToString();
85  if (i != mappers_.size() - 1) {
86  ret += ", ";
87  }
88  i++;
89  }
90  ret += "}";
91  }
92  ret += "]";
93  }
94  return ret;
95 }

101 void Type::AddMapper(const std::shared_ptr<TypeMapper> &mapper, bool remove_existing) {
102  Type *other = mapper->b();
103  // Check if a mapper doesn't already exists
104  if (GetMapper(other, false)) {
105  if (!remove_existing) {
106  CERATA_LOG(FATAL, "Mapper already exists to convert "
107  "from " + this->ToString(true, true) + " to " + other->ToString(true, true));
108  } else {
109  RemoveMappersTo(other);
110  }
111  }
112 
113  // Check if the supplied mapper does not convert from this type.
114  if (mapper->a() != this) {
115  CERATA_LOG(FATAL, "Type converter does not convert from " + name());
116  }
117 
118  // Add the converter to this Type
119  mappers_.push_back(mapper);
120 
121  // If the other type doesnt already have it, add the inverse map there as well.
122  if (!other->GetMapper(this)) {
123  other->AddMapper(mapper->Inverse());
124  }
125 }

131 std::optional<std::shared_ptr<TypeMapper>> Type::GetMapper(Type *other, bool generate_implicit) {
132  // Search for an existing type mapper.
133  for (const auto &m : mappers_) {
134  if (m->CanConvert(this, other)) {
135  return m;
136  }
137  }
138 
139  if (generate_implicit) {
140  // Implicit type mappers maybe be generated in three cases:
141 
142  // If it's exactly the same type object,
143  if (other == this) {
144  // Generate a type mapper to itself using the TypeMapper constructor.
145  return TypeMapper::Make(this);
146  }
147 
148  // If there is an explicit function in this Type to generate a mapper:
149  if (CanGenerateMapper(*other)) {
150  // Generate, add and return the new mapper.
151  auto new_mapper = GenerateMapper(other);
152  this->AddMapper(new_mapper);
153  return new_mapper;
154  }
155 
156  // Or if its an "equal" type, where each flattened type is equal.
157  if (IsEqual(*other)) {
158  // Generate an implicit type mapping.
159  return TypeMapper::MakeImplicit(this, other);
160  }
161  }
162 
163  // There is no mapper
164  return {};
165 }

167 int Type::RemoveMappersTo(Type *other) {
168  int removed = 0;
169  for (auto m = mappers_.begin(); m < mappers_.end(); m++) {
170  if ((*m)->CanConvert(this, other)) {
171  mappers_.erase(m);
172  removed++;
173  }
174  }
175  return removed;
176 }

384 bool Stream::CanGenerateMapper(const Type &other) const {
385  switch (other.id()) {
386  case Type::STREAM:
387  // If this and the other streams are "equal", a mapper can be generated
388  if (IsEqual(other)) {
389  return true;
390  } else {
391  // We can also map an empty stream, without mapping the elements. In practise, a back-end might emit e.g.
392  // additional ready/valid/count wires, connect those, but not any data elements.
393  if ((this->element_type() == nul()) || (dynamic_cast<const Stream &>(other).element_type() == nul())) {
394  return true;
395  }
396  }
397  default:return false;
398  }
399 }

401 std::shared_ptr<TypeMapper> Stream::GenerateMapper(Type *other) {
402  // Check if we can even do this:
403  if (!CanGenerateMapper(*other)) {
404  CERATA_LOG(FATAL, "No mapper generator known from Stream to " + other->name() + ToString(other->id()));
405  }
406  if (IsEqual(*other)) {
407  return TypeMapper::MakeImplicit(this, other);
408  } else {
409  // If this or the other stream has a no element type:
410  if ((this->element_type() == nul()) || (dynamic_cast<Stream *>(other)->element_type() == nul())) {
411  auto mapper = TypeMapper::Make(this, other);
412  // Only connect the two stream flat types.
413  auto matrix = mapper->map_matrix();
414  matrix(0, 0) = 1;
415  mapper->SetMappingMatrix(matrix);
416  return mapper;
417  }
418  }
419  return nullptr;
420 }

422 bool Record::IsEqual(const Type &other) const {
423  if (&other == this) {
424  return true;
425  }
426  // Must also be a record
427  if (!other.Is(Type::RECORD)) {
428  return false;
429  }
430  // Must have same number of fields
431  auto &other_record = dynamic_cast<const Record &>(other);
432  if (other_record.num_fields() != this->num_fields()) {
433  return false;
434  }
435  // Each field must also be of equal type
436  for (size_t i = 0; i < this->num_fields(); i++) {
437  auto a = this->field(i)->type();
438  auto b = other_record.field(i)->type();
439  if (!a->IsEqual(*b)) {
440  return false;
441  }
442  }
443  // If we didn't return already, the Record Types are the same
444  return true;
445 }
