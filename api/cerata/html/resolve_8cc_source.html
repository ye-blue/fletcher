32 static void InsertSignal(Component *comp, Edge *edge, std::deque<Object *> *resolved) {
33  CERATA_LOG(DEBUG, "VHDL: Resolving " + edge->src()->ToString() + " --> " + edge->dst()->ToString());
34  std::string prefix;
35  if (edge->src()->parent()) {
36  prefix = edge->src()->parent().value()->name() + "_";
37  } else if (edge->dst()->parent()) {
38  prefix = edge->dst()->parent().value()->name() + "_";
39  }
40  // Remember we've touched these nodes already
41  resolved->push_back(edge->src());
42  resolved->push_back(edge->dst());
43  // Insert a new signal, after this, the original edge may potentially be destroyed.
44  auto sig = insert(edge, prefix, comp);
45 }

47 static void ResolvePorts(Component *comp, Instance *inst, std::deque<Object *> *resolved) {
48  // First get all normal ports.
49  for (const auto &port : inst->GetAll<Port>()) {
50  // Iterate over all edges that are sinks or sources of this port.
51  for (const auto &edge : port->sinks()) {
52  // If the destination is not a port, continue
53  if (!edge->dst()->IsPort()) { continue; }
54  // If the source or destination is a component node on the component we're working on,
55  // rather than an instance node, continue.
56  if ((edge->src()->parent() == comp) || (edge->dst()->parent() == comp)) { continue; }
57  // If the destination is already resolved, continue.
58  if (Contains(*resolved, dynamic_cast<Object *>(edge->dst()))) { continue; }
59  // We are now sure to be dealing with a port-to-port connection on two instances.
60  // VHDL cannot handle port-to-port connections of instances.
61  // Insert a signal in between and add it to the component.
62  InsertSignal(comp, edge, resolved);
63  }
64  }
65 }

67 static void InsertSignalArray(Component *comp, Edge *edge, std::deque<Object *> *resolved) {
68  auto port = dynamic_cast<Port *>(edge->src());
69  if (!port->array()) {
70  throw std::runtime_error("Expected edge source to be child of NodeArray.");
71  }
72  auto na = port->array().value();
73  auto pa = dynamic_cast<const PortArray *>(na);
74  if (pa == nullptr) {
75  throw std::runtime_error("Edge source is child of NodeArray, but not PortArray.");
76  }
77  if (!pa->parent()) {
78  throw std::runtime_error("PortArray has no parent.");
79  }
80 
81  CERATA_LOG(DEBUG, "VHDL: Resolving port array " + pa->name() + " --> " + edge->dst()->ToString());
82 
83  std::string prefix;
84  if (edge->src()->parent()) {
85  prefix = edge->src()->parent().value()->name() + "_";
86  } else if (edge->dst()->parent()) {
87  prefix = edge->dst()->parent().value()->name() + "_";
88  }
89 
90  auto sig_name = prefix + port->name();
91  auto size_name = sig_name + "_" + pa->size()->name();
92 
93  // Make a copy of the size node.
94  auto size_node = std::dynamic_pointer_cast<Node>(pa->size()->Copy());
95  size_node->SetName(size_name);
96 
97  // Create a new SignalArray.
98  // This array is going to live on the component graph, so we also need a copy of the size node, since the current
99  // size node lives on the instance graph only. We can only use an instance parameter from top to bottom, not vice
100  // versa in VHDL.
101  auto sa = SignalArray::Make(sig_name, port->type()->shared_from_this(), size_node, port->domain());
102 
103  // Add the SignalArray to the component.
104  comp->Add(sa);
105 
106  // Now for all nodes on the PortArray.
107  for (size_t n = 0; n < pa->num_nodes(); n++) {
108  // Create a new child node inside the signal array.
109  // Don't increment the size, since we can just share the existing size node.
110  auto sn = sa->Append();
111  // All sinks of this port should now be sourced by the signal array child node.
112  for (auto &e : pa->node(n)->sinks()) {
113  // Get the destination node.
114  auto dn = e->dst();
115  Connect(dn, sn);
116  // Mark both ends as resolved.
117  resolved->push_back(dn);
118  resolved->push_back(sn);
119  // Remove the original edge from the port array child node.
120  pa->node(n)->RemoveEdge(e);
121  }
122  // Connect the port array node as source to destination signal array node.
123  Connect(sn, pa->node(n));
124  }
125 }

127 static void ResolvePortArrays(Component *comp, Instance *inst, std::deque<Object *> *resolved) {
128  // First get all port arrays.
129  for (const auto &pa : inst->GetAll<PortArray>()) {
130  // We first figure out if a port array child node has a connection to another instance port.
131  for (const auto &pac : pa->nodes()) {
132  for (const auto &edge : pac->sinks()) {
133  // If either side is not a port, continue with the next edge.
134  if (!edge->dst()->IsPort()) { continue; }
135  // If either sides of the edges are a port node on this component, continue, since this is allowed in VHDL.
136  if ((edge->src()->parent() == comp) || (edge->dst()->parent() == comp)) { continue; }
137  // If the destination is already resolved, continue.
138  if (Contains(*resolved, dynamic_cast<Object *>(edge->src()))) { continue; }
139  if (Contains(*resolved, dynamic_cast<Object *>(edge->dst()))) { continue; }
140  // We are now sure we are dealing with a PortArray child - to - port.
141  // InsertSignalArray will solve this for us.
142  InsertSignalArray(comp, edge, resolved);
143  }
144  }
145  }

148 Component *Resolve::ResolvePortToPort(Component *comp) {
149  std::deque<Object *> resolved;
150  CERATA_LOG(DEBUG, "VHDL: Resolving port-to-port connections...");
151  for (const auto &inst : comp->children()) {
152  ResolvePorts(comp, inst, &resolved);
153  ResolvePortArrays(comp, inst, &resolved);
154  }
155  CERATA_LOG(DEBUG, "VHDL: Resolved " + std::to_string(resolved.size()) + " port-to-port connections...");
156  return comp;
157 }

186 static void ExpandStreamType(Type *type) {
187  // If the type was not expanded yet.
188  if (type->meta.count(metakeys::WAS_EXPANDED) == 0) {
189  // Expand it. First remember that it was expanded.
190  type->meta[metakeys::WAS_EXPANDED] = "true";
191  // Flatten the type.
192  auto flattened_types = Flatten(type);
193  // Check if it has any streams.
194  if (HasStream(flattened_types)) {
195  CERATA_LOG(DEBUG, "VHDL: Expanding type " + type->ToString());
196  // Iterate over the flattened type
197  for (auto &flat : flattened_types) {
198  // If we encounter a stream type
199  if (flat.type_->Is(Type::STREAM)) {
200  auto *stream_type = dynamic_cast<Stream *>(flat.type_);
201  // Check if we didn't expand the type already.
202  if (stream_type->meta.count(metakeys::EXPAND_TYPE) == 0) {
203  // Create a new record to hold valid, ready and the original element type
204  auto new_elem_type = Record::Make(stream_type->name() + "_vr");
205  // Mark the record
206  new_elem_type->meta[metakeys::EXPAND_TYPE] = "record";
207  // Add valid, ready and original type to the record and set the new element type
208  new_elem_type->AddField(RecField::Make("valid", valid()));
209  new_elem_type->AddField(RecField::Make("ready", ready(), true));
210  new_elem_type->AddField(RecField::Make(stream_type->element_name(), stream_type->element_type()));
211  stream_type->SetElementType(new_elem_type);
212  // Mark the stream itself to remember we've expanded it.
213  stream_type->meta[metakeys::EXPAND_TYPE] = "stream";
214  }
215  }
216  }
217  }
218  }

221 static void ExpandMappers(Type *type, const std::deque<std::shared_ptr<TypeMapper>> &mappers) {
222  // TODO(johanpel): Generalize this type expansion functionality and add it to Cerata.
223  // Iterate over all previously existing mappers.
224  for (const auto &mapper : mappers) {
225  // Skip mappers that don't contain streams.
226  if (!HasStream(mapper->flat_a()) && !HasStream(mapper->flat_b())) {
227  continue;
228  }
229  // Skip mappers that were already expanded.
230  if (mapper->meta.count(metakeys::WAS_EXPANDED) > 0) {
231  continue;
232  }
233 
234  // Get a copy of the old matrix.
235  auto old_matrix = mapper->map_matrix();
236  // Create a new mapper
237  auto new_mapper = TypeMapper::Make(type, mapper->b());
238  // Get the properly sized matrix
239  auto new_matrix = new_mapper->map_matrix();
240  // Get the flattened type
241  auto flat_a = new_mapper->flat_a();
242  auto flat_b = new_mapper->flat_b();
243 
244  int64_t old_row = 0;
245  int64_t old_col = 0;
246  int64_t new_row = 0;
247  int64_t new_col = 0;
248 
249  while (new_row < new_matrix.height()) {
250  auto at = flat_a[new_row].type_;
251  while (new_col < new_matrix.width()) {
252  auto bt = flat_b[new_col].type_;
253  // Figure out if we're dealing with a matching, expanded type on both sides.
254  if (IsExpandType(at, "stream") && IsExpandType(bt, "stream")) {
255  new_matrix(new_row, new_col) = old_matrix(old_row, old_col);
256  new_col += 4; // Skip over record, valid and ready
257  old_col += 1;
258  } else if (IsExpandType(at, "record") && IsExpandType(bt, "record")) {
259  new_matrix(new_row, new_col) = old_matrix(old_row, old_col);
260  new_col += 3; // Skip over valid and ready
261  old_col += 1;
262  } else if (IsExpandType(at, "valid") && IsExpandType(bt, "valid")) {
263  new_matrix(new_row, new_col) = old_matrix(old_row, old_col);
264  new_col += 2; // Skip over ready
265  old_col += 1;
266  } else if (IsExpandType(at, "ready") && IsExpandType(bt, "ready")) {
267  new_matrix(new_row, new_col) = old_matrix(old_row, old_col);
268  new_col += 1;
269  } else {
270  // We're not dealing with a *matching* expanded type. However, if the A side was expanded and the type is
271  // not matching, we shouldn't make a copy. That will happen later on, on another row.
272  if (!IsExpandType(at)) {
273  new_matrix(new_row, new_col) = old_matrix(old_row, old_col);
274  }
275  new_col += 1;
276  }
277  // If this column was not expanded, or if it was the last expanded column, move to the next column in the old
278  // matrix.
279  if (!IsExpandType(bt) || IsExpandType(bt, "ready")) {
280  old_col += 1;
281  }
282  }
283  old_col = 0;
284  new_col = 0;
285 
286  // Only move to the next row of the old matrix when we see the last expanded type or if it's not an expanded
287  // type at all.
288  if (!IsExpandType(at) || IsExpandType(at, "ready")) {
289  old_row += 1;
290  }
291  new_row += 1;
292  }
293 
294  // Set the mapping matrix of the new mapper to the new matrix
295  new_mapper->SetMappingMatrix(new_matrix);
296  new_mapper->meta[metakeys::WAS_EXPANDED] = "true";
297 
298  // Add the mapper to the type
299  type->AddMapper(new_mapper);
300  }
301 }

303 Component *Resolve::ExpandStreams(Component *comp) {
304  CERATA_LOG(DEBUG, "VHDL: Materialize stream abstraction...");
305  // List of types.
306  std::deque<Type *> types;
307  // List of mappers. Takes ownership of existing mappers before expansion.
308  std::unordered_map<Type *, std::deque<std::shared_ptr<TypeMapper>>> mappers;
309 
310  // Populate the list.
311  GetAllTypes(comp, &types, true);
312 
313  // First expand all the types.
314  for (const auto &type : types) {
315  auto type_mappers = type->mappers();
316  if (!type_mappers.empty()) {
317  mappers[type] = type_mappers; // Remember and keep the old mappers for ExpandMappers
318  }
319  ExpandStreamType(type);
320  }
321  // Next, expand the mappers that were present for all types.
322  for (const auto &type : types) {
323  if (mappers.count(type) > 0) {
324  ExpandMappers(type, mappers[type]);
325  }
326  }
327 
328  return comp;
329 }
