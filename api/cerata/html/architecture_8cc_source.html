25 MultiBlock Arch::Generate(const Component &comp) {
26  MultiBlock ret;
27  Line header_line;
28  header_line << "architecture Implementation of " + comp.name() + " is";
29  ret << header_line;
30 
31  // Component declarations
32  auto components_used = comp.GetAllUniqueComponents();
33  for (const auto &c : components_used) {
34  // Check for metadata that this component is not marked primitive
35  // In this case, a library package has been added at the top of the design file.
36  if ((c->meta().count(metakeys::PRIMITIVE) == 0) || (c->meta().at(metakeys::PRIMITIVE) != "true")) {
37  // TODO(johanpel): generate packages with component declarations
38  auto comp_decl = Decl::Generate(*c);
39  ret << comp_decl;
40  }
41  }
42 
43  // Signal declarations
44  auto signals = comp.GetAll<Signal>();
45  for (const auto &s : signals) {
46  auto signal_decl = Decl::Generate(*s, 1);
47  ret << signal_decl;
48  }
49 
50  // Signal array declarations
51  auto signal_arrays = comp.GetAll<SignalArray>();
52  for (const auto &s : signal_arrays) {
53  auto signal_array_decl = Decl::Generate(*s, 1);
54  ret << signal_array_decl;
55  }
56 
57  Line header_end;
58  header_end << "begin";
59  ret << header_end;
60 
61  // Component instantiations
62  auto instances = comp.children();
63  for (const auto &i : instances) {
64  auto inst_decl = Inst::Generate(*i);
65  ret << inst_decl;
66  }
67 
68  // TODO(johanpel): Signal connections
69 
70  Line footer;
71  footer << "end architecture;";
72  ret << footer;
73 
74  return ret;
75 }
