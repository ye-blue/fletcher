32 std::shared_ptr<const Node> Expression::MergeIntLiterals(const Expression *exp) {
33  if (exp->lhs_->IsLiteral() && exp->rhs_->IsLiteral()) {
34  auto l = std::dynamic_pointer_cast<const Literal>(exp->lhs_);
35  auto r = std::dynamic_pointer_cast<const Literal>(exp->rhs_);
36  if ((l->storage_type() == Literal::StorageType::INT)
37  && (r->storage_type() == Literal::StorageType::INT)
38  && (l->type() == r->type())) {
39  std::shared_ptr<Node> new_node;
40  switch (exp->operation_) {
41  case Op::ADD: return intl(l->IntValue() + r->IntValue());
42  case Op::SUB: return intl(l->IntValue() - r->IntValue());
43  case Op::MUL: return intl(l->IntValue() * r->IntValue());
44  case Op::DIV: return intl(l->IntValue() / r->IntValue());
45  }
46  }
47  }
48  return exp->shared_from_this();
49 }

51 std::shared_ptr<const Node> Expression::EliminateZeroOne(const Expression *exp) {
52  switch (exp->operation_) {
53  case Op::ADD: {
54  if (exp->lhs_ == intl(0)) return exp->rhs_;
55  if (exp->rhs_ == intl(0)) return exp->lhs_;
56  break;
57  }
58  case Op::SUB: {
59  if (exp->rhs_ == intl(0)) return exp->lhs_;
60  break;
61  }
62  case Op::MUL: {
63  if (exp->lhs_ == intl(0)) return intl(0);
64  if (exp->rhs_ == intl(0)) return intl(0);
65  if (exp->lhs_ == intl(1)) return exp->rhs_;
66  if (exp->rhs_ == intl(1)) return exp->lhs_;
67  break;
68  }
69  case Op::DIV: {
70  if (exp->lhs_ == intl(0) && exp->rhs_ != intl(0)) return intl(0);
71  if (exp->rhs_ == intl(0)) { CERATA_LOG(FATAL, "Division by 0."); }
72  if (exp->rhs_ == intl(1)) return exp->lhs_;
73  break;
74  }
75  }
76  return exp->shared_from_this();
77 }

79 std::shared_ptr<const Node> Expression::Minimize(const Node *node) {
80  std::shared_ptr<const Node> result = node->shared_from_this();
81 
82  // If this node is an expression, we need to mimize its lhs and rhs first.
83  if (node->IsExpression()) {
84  auto expr = std::dynamic_pointer_cast<const Expression>(result);
85  // Attempt to minimize children.
86  auto min_lhs = Minimize(expr->lhs());
87  auto min_rhs = Minimize(expr->rhs());
88 
89  // If minimization took place in either node, create a new expression with the minimized nodes.
90  if ((min_lhs != expr->lhs_) || (min_rhs != expr->rhs_)) {
91  expr = Expression::Make(expr->operation_, min_lhs, min_rhs);
92  }
93 
94  // Apply zero/one elimination on the minimized expression.
95  result = EliminateZeroOne(expr.get());
96 
97  // Integer literal merging
98  if (result->IsExpression()) {
99  expr = std::dynamic_pointer_cast<const Expression>(result);
100  result = MergeIntLiterals(expr.get());
101  }
102  // TODO(johanpel): put some more elaborate minimization function/rules etc.. here
103  }
104  return result;
105 }

117 std::string Expression::ToString() const {
118  auto min = Minimize(this);
119  if (min->IsExpression()) {
120  auto mine = std::dynamic_pointer_cast<const Expression>(min);
121  auto ls = mine->lhs_->ToString();
122  auto op = cerata::ToString(operation_);
123  auto rs = mine->rhs_->ToString();
124  return ls + op + rs;
125  } else {
126  return min->ToString();
127  }
