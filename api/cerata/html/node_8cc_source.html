59 bool MultiOutputNode::AddEdge(const std::shared_ptr<Edge> &edge) {
60  bool success = false;
61  // Check if this edge has a source
62  if (edge->src()) {
63  // Check if the source is this node
64  if (edge->src() == this) {
65  // Check if this node does not already contain this edge
66  if (!Contains(outputs_, edge)) {
67  // Add the edge to this node
68  outputs_.push_back(edge);
69  success = true;
70  }
71  }
72  }
73  return success;
74 }

76 bool MultiOutputNode::RemoveEdge(Edge *edge) {
77  if (edge->src()) {
78  if (edge->src() == this) {
79  // This node sources the edge.
80  for (auto i = outputs_.begin(); i < outputs_.end(); i++) {
81  if (i->get() == edge) {
82  outputs_.erase(i);
83  return true;
84  }
85  }
86  }
87  }
88  return false;
89 }

110 bool NormalNode::AddEdge(const std::shared_ptr<Edge> &edge) {
111  // first attempt to add the edge as an output.
112  bool success = MultiOutputNode::AddEdge(edge);
113  // If we couldn't add the edge as output, it must be input.
114  if (!success) {
115  // Check if the edge has a destination.
116  if (edge->dst()) {
117  // Check if this is the destination.
118  if (edge->dst() == this) {
119  // Set this node source to the edge.
120  input_ = edge;
121  success = true;
122  }
123  }
124  }
125  return success;
126 }

128 bool NormalNode::RemoveEdge(Edge *edge) {
129  // First remove the edge from any outputs
130  bool success = MultiOutputNode::RemoveEdge(edge);
131  // Check if the edge is an input to this node
132  if (edge->dst() && !success) {
133  if ((edge->dst() == this) && (input_.get() == edge)) {
134  input_.reset();
135  success = true;
136  }
137  }
138  return success;
139 }
